在开始之前，我们定义一些特殊的数据：

//这里设计的是16位数据总线和16位地址总线的CPU

//CPU运行时的判断指令

#define flase 00b //定义flase为0

#define true 01b //定义true为1

#define enable 01b //定义高电平为1

#define disable 00b //定义低电平为0

//定义寄存器

#define NULL 0000b //无寄存器操作

#define dateremOne 0001b //数据寄存器1

#define dateremtwo 0010b //数据寄存器2

#define adressrem 0011b //地址寄存器

#define ESP 0100b //指针寄存器，当然为了简便这里不详细去实现了

//译码后的判断指令定义

#define nop 0000b //定义NOP指令

#define mov 0001b //定义mov指令

#define add 0010b //定义加法指令

#define sub 0011b //定义减法指令

#define mul 0100b //定义乘法指令

#define div 0101b //定义除法指令

#define jmp 0110b //定义跳转指令

#define push 0111b //定义压栈指令

#define pop 1000b //定义弹栈指令

////还可以多加几个但是这里就不加了，这几个最基本的汇编指令够用了，后面如果想再实现其它功能时再加，这里进行定义是为了我们更好的用编程语言描述，不然一堆二进制码我们是没法读的，也不方便进行流程说明，后面具体涉及到数字电路时咱们就必须用二进制说明了，用1代表高电平，用0代表低电平（也就是开关状态）。

//CPU从最小结构到最大机构数据建模

//指令码也是16位，前四位为操作码，中间四位为寄存器选择符，最后八位为地址码或者实数

Struct DeCode { //指令译码器

Input wire CodeLine[3:0];//输入指令位数

Input wire Rem[3:0];//寄存器操作码

Input wire Address[7:0];//地址码或者实数，这个地址码是偏移地址

Output wire OutRem[3:0];//寄存器操作数输出

Output wire OutCodeLine[3:0];//输出指令

Output wire OutAddress[7:0];//地址和实数输出

}

Struct CodeRem{ //指令寄存器

Input wire indate[15:0];//寄存器输入

Output wire outdate[15:0];//寄存器输出

}

Struct CPUControl { //控制单元

Input wire ControlLine[15:0];//读取操作码

Output wire outCodeLine[15:0];//指令输出

Inout wire Address[15:0];//地址输入输出端口

BIT PCAddress[16]; //指令计数器中保存着指令地址值

CodeRem CodeRemMod;

DeCode DeCodeMod;

}

Struct CPU{

//CPU有很多引脚我们这里只用到几个就行

Input wire rst;//复位引脚

Input wire Vcc;//电源引脚

Input wire Enable;//使能引脚，只有此引脚输入高电平时，CPU才能正常运行

Output wire Gnd;//接地引脚，接地很重要，防止芯片内部电压过高击穿电子器件当然这个引脚不进行数据传输只是保证芯片正常工作用的

Input wire clk;//时钟引脚

Inout wire Adress[15:0];//地址选取端口

Input wire Code[15:0];//指令读取端口

Input wire Date[15:0];//数据输入端口，也可以把数据输入输出端口合并

Output wire Outdate[15:0];//数据输出端口

CPUControl CPUControlMod;//控制单元

CPUDateRem CPUDateRemMod;//数据寄存器单元

CPUMath CPUMathMod;//运算单元

Bit [] CodeLine[4]; //操作码存放寄存器

Bit [] Rem[4];//寄存器选择符寄存器

//首先进行导线连接，这些部分本来是在各个模块当中进行，但这里单独拿出来进行连接这样有助于我们理解CPU的实际线路走向和工作流程，到具体实现CPU数字电路时不会头晕，其实这几期比较枯燥，小编尽量把枯燥的东西讲详细点，后面小编会画一下具体的流程图

//首先通过读取指令引线到控制单元，控制单元的指令码要送入译码器用 "=>" 代表把引线连接起来当然不同的硬件编程语言可能表述方式不一样，电路跟水路有点像是有流向的！而且数字电路二级管是严格的流向控制的，反向有高阻状态

CPU () {

////////取指令阶段

//CPU地址选址

Address => CPUControlMod.Address;//地址传输到控制单元

//在CPU初始化时程序的地址已经被BIOS改变成程序入口点

CPUControlMod.PCAddress = CPUControlMod.Address;//把程序地址保存到指令计数器当中，当然这个计数器是简化的，还可以扩展

Code => CPUControlMod.ControlLine; //读取指令后数据流向控制单元，

CPUControlMod.ControlLine => CPUControlMod.outCodeLine => CPUControlMod.CodeRem.indate;//指令流向指令寄存器这里不直接连输出口，是因为咱们这个CPU还可以扩展功能，现在实现的是基本的模块

CPUControlMod.CodeRem.indate => CPUControlMod.CodeRem.outdate;//指令从寄存器中输出咱们为了方便把寄存过程省略了

////////指令译码阶段

CPUControlMod.CodeRem.outdate[3:0] => CPUControlMod.DeCode.CodeLine;//这里开始指令操作码与地址码分开

CPUControlMod.CodeRem.outdate[7:4] => CPUControlMod.DeCode.Rem;//寄存器选择码

CPUControlMod.DeCode.Rem => CPUControlMod.DeCode.OutRem; //寄存器操作码输出

CPUControlMod.CodeRem.outdate[15:8] => CPUControlMod.DeCode.Address;//得到偏移地址

CPUControlMod.DeCode.CodeLine => CPUControlMod.DeCode.OutCodeLine;//把操作码传输出指令寄存器

CPUControlMod.DeCode.addressDate = 0x00h || CPUControlMod.DeCode.Address;//把地址扩展这个可以放到算术逻辑单元当中计算，这里为了方便直接计算了

CPUControlMod.DeCode.addressDate => CPUControlMod.DeCode.OutAddress ;//把偏移地址地址值传输

//其实在寄存器当中也是可以实现部分逻辑指令，但这里小编尽量简化，咱们只是要了解CPU结构，但太复杂的话后面也不利于我们电路讲解，而且小编是真的全凭脑袋回忆和原理来实现的，太费神的话小编也吃不消。

//控制单元连线完毕

//接下来是整个CPU控制和处理流程，这里的连线比较错综复杂不能用直连方式表达

CodeLine = CPUControlMod.DeCode.OutCodeLine;//CPU与操作码有关

Rem = CPUControlMod.DeCode.OutRem;//寄存器选择符

//////执行指令阶段

if (Enable == enable){

If(rst == deable){

If(clk == enable){

Switch (Rem){

Break;

Case dateremOne:

//将数据传输给数据寄存器1中

CPUControlMod.DeCode.OutAddress => CPUDateRemMod. Indateone;

CPUDateRemMod.dateOne = CPUDateRemMod. Indateone;

Break;

Case dateremtwo:

//将数据传输给数据寄存器2中

CPUControlMod.DeCode.OutAddress => CPUDateRemMod. Indatetwo;

CPUDateRemMod.dateTwo = CPUDateRemMod. Indatetwo;

Break;

Case adressrem:

//将地址或者数据传到地址寄存器当中，这里不实现了

Break;

Break;

}

Break;

Case add:

//加法操作

Switch (Rem){

Case NULL:

//空操作

Break;

Case dateremOne:

//用数据寄存器1的数据与实数或地址相加结果保存到数据寄存器1中

//将译码器当中的数据传输给运算单元

CPUControlMod.DeCode.OutAddress => CPUMathMod.inDatetwo;

//将数据寄存器1中的数据传送到运算单元中

CPUDateRemMod.dateOne => CPUDateRemMod.outdateone;

CPUDateRemMod.outdateone => CPUMathMod.inDateone;

//进行计算，其实中间最好有个过程但这里就不实现了

CPUMathMod.outDate = CPUMathMod.inDateone + CPUMathMod.inDatetwo;

//运算结果保存到数据存器1中

Break;

Case dateremtwo:

//用数据寄存器2的数据与实数或地址相加结果保存到数据寄存器2中

//将译码器当中的数据传输给运算单元

//将数据寄存器2中的数据传送到运算单元中
