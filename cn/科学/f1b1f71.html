Python循序渐近：字符串类型

2020-10-29
| 科学

# 字符串类型是不可变序列

# 字符串类型通过字符串字面值和构造器创建

# 字面值三种形式：单引号、双引号和三引号

# 构造器：str(object='')

# 构造器：str(object=b'', encoding='utf-8', errors='strict')

str1 = "Hello"

str2 = 'World!'

str3 = """Welcome to

China!"""

print(type(str1))

print(str1)

print(str2)

print(str3)

str4 = str() # str = ""

print(str4)

str5 = str("Hello, kitty!")

print(str5)

# 以下为方法演示

demo_str = "hello World!"

# 首个字符大写，其余字符为小写

print(demo_str.capitalize())

# 转换为小写，与lower()相比，功能更强大，会转换ASCII码以外的大小写

print(demo_str.casefold())

# 返回指定长度的字符串，原字符串在其中，两边填充指定字符串，若省略，则填充空白

print(demo_str.center(50, "="))

# 返回指定字符串出现的次数

print(demo_str.count("o", 0, 20))

# 编码成字节串对象

print(demo_str.encode())

# 判定字符串是否以指定字符串结尾

print(demo_str.endswith("!"))

# 替换制表符

print(demo_str.expandtabs(tabsize=1))

# 返回指定字符串在字符串中的索引

print(demo_str.find("lo", 0, 100))

# 类似于find()，但index()找不到指定字符串会引发ValueError

# ValueError: substring not found

# print(demo_str.index("HHH", 0, 100))

print(demo_str.isalnum())

print(demo_str.isalpha())

print(demo_str.isascii())

print(demo_str.isdecimal())

print(demo_str.isdigit())

print(demo_str.isidentifier())

print(demo_str.islower())

print(demo_str.isnumeric())

print(demo_str.isprintable())

print(demo_str.isspace())

print(demo_str.istitle())

print(demo_str.isupper())

# 字符串拼接

print(demo_str.join("abc"))

# 返回长度为 width 的字符串，原字符串在其中靠左对齐

print(demo_str.ljust(50, "-"))

# 返回原字符串的副本，其所有区分大小写的均转换为小写

print(demo_str.lower())

# 返回原字符串的副本，移除其中的前导字符

print(demo_str.lstrip())

# 在sep首次出现的位置拆分字符串，返回一个3元组，其中包含分隔符之前的部分、分隔符本身，以及分隔符之后的部分

print(demo_str.partition("r"))

# 返回字符串的副本，其中出现的所有子字符串old都将被替换为new

print(demo_str.replace("o", "0"))

print(demo_str.rfind("H"))

print(demo_str.rindex("h"))

print(demo_str.rjust(50, "%"))

print(demo_str.rpartition("r"))

print(demo_str.rsplit())

print(demo_str.split())

print(demo_str.rstrip())

print(demo_str.splitlines())

print(demo_str.startswith("h"))

print(demo_str.swapcase())

print(demo_str.title())

# print(demo_str.maketrans())

# print(demo_str.translate())

print(demo_str.zfill(20))
