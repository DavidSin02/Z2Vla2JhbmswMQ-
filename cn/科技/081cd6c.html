void trimSpace(char *str)

{

int i, j, k;

char tmpstr[MAXSTRSIZE]; // 此处用户可动态分配，满足情变

i = strlen(str);

for (k = j = 0; j<i; j++)

{

if (str[j] != ' ')

{

tmpstr[k] = str[j]; k++;

}

}

tmpstr[k] = '\0';

strcpy(str, tmpstr);

}
