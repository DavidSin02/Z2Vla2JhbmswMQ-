100行代码实现一个简单有趣的记录鼠标走过距离的小程序

2020-10-29
| 科技

100行代码实现一个简单有趣的记录鼠标走过距离的小程序，稍加修改可以记录鼠标走过的位置和所作的操作

记着那时候的我们还在捣鼓Windows下的软件开发，模仿使用MFC写程序，奈何MFC的消息驱动、宏、虚函数，层层继承，整个框架定义的实在是太绕了，搞不清楚到底是如何运作的，消息函数怎么关联，何时派发等等，实在是摸不清楚门道。找了一段时间后发现Windows下也可以通过原生Win32编程，而MFC则是Win32编程的抽象和封装。

一个窗口程序，分为2部分，窗口UI和对应的消息处理逻辑。

1、操作系统会提供一些UI组件，比如按钮，文本框，进度条等等，程序只要调用系统API就可以在显示器上画出UI的图形。

2、整个Windows程序是基于消息驱动的。操作系统负责管理外设（键盘、鼠标），统一管理各种请求，并负责将这些输入/请求，转化为一个通知，告诉应用程序某个事情发生了。比如单击鼠标、改变窗口尺寸、按下键盘上的一个键。应用程序收到消息后就要处理。

Windows下的消息型如下图。系统会给每一个程序创建一个队列用来存放消息结构体，每产生1个消息就会增加到队尾。应用程序则有一个死循环不断的从队头读消息，并调用用户定义的函数处理。队列中没有消息时就等待，整个过程就这么简单。


Windows消息处理示意图

WinMain是入口函数，通过DialogBox API，按照IDD_DIALOG1模板创建了一个对话框UI，并将WndDlg作为消息处理函数

Wint WINAPI WinMain(HINSTANCE hInstance,HINSTANCE hPreInstance,LPSTR lpCmdLine, int nShowCmd){	hInst=hInstance;	if (__argc == 1)						DialogBox(hInst,"IDD_DIALOG1",NULL,WndDlg);     //DialogBox封装了窗口UI，WndDlg是消息处理函数	else																																							//也就是示意图中的WndProc	{		HANDLE hProcessOrig = (HANDLE) _ttoi(__targv[1]);		WaitForSingleObject(hProcessOrig, INFINITE);		CloseHandle(hProcessOrig);		DeleteFile(__targv[2]);	} 	return 1;}

系统会定义很多消息，每个消息对应不同的处理，可以通过Spy++工具查看消息。消息处理函数

BOOL CALLBACK WndDlg(HWND hwnd,UINT msg,WPARAM wParam,LPARAM lParam){	static	POINT point1,point2;	int l;	static unsigned int size=100;	static double len=0;	TCHAR str[120];	static HDC hdc;	static int xscr,yscr;	static BOOL chuang=FALSE;	PROCESS_INFORMATION  pi;	STARTUPINFO si;	TCHAR szPathOrig[_MAX_PATH], szPathClone[_MAX_PATH],szCmdLine[512]; 	HANDLE hfile ;	HANDLE hProcessOrig;	switch(msg)	{	case WM_INITDIALOG:												#窗口还没有显示出来前，让用户有机会进行一些初始化操作		l=GetSystemMetrics(SM_CYSCREEN);		wsprintf(str,"屏幕宽: %d",l);		SetWindowText(GetDlgItem(hwnd,IDC_KUAN),str);		l=GetSystemMetrics(SM_CXSCREEN);		wsprintf(str,"屏幕长: %d",l);		SetWindowText(GetDlgItem(hwnd,IDC_CHANG),str);		hdc=GetDC((HWND)0);		SetMapMode(hdc,MM_LOMETRIC);		SetTimer(hwnd,ID_TIMER,100,NULL);           //定时器为0.1s		GetCursorPos(&point1);		xscr=GetSystemMetrics(SM_CXSCREEN);		yscr=GetSystemMetrics(SM_CYSCREEN);		CheckDlgButton(hwnd,IDC_RADIO1,BST_CHECKED);		return TRUE;	case WM_TIMER:                              //通过定时器消息获取每次鼠标的位置		GetCursorPos(&point2);		len=len+abs(int(sqrt((float)(point2.x-point1.x)*(point2.x-point1.x)+(point2.y-point1.y)*(point2.y-point1.y))));		//			_gcvt( len/size, 3, str );		sprintf(str,"%lf",len/size );		SetWindowText(GetDlgItem(hwnd,IDC_MOUSELEN),str);		point1.x=point2.x, point1.y=point2.y;		if(chuang==TRUE)		{							if(point1.x>xscr-2) point1.x=2;			if(point1.x<2)	point1.x=xscr-2;			if(point1.y>yscr-2) point1.y=2;			if(point1.y<2)	point1.y=yscr-2;			SetCursorPos(point1.x,point1.y);		}		return TRUE;	case WM_COMMAND:		switch(LOWORD(wParam))		{		case IDC_BUTTON1:                   //实现删除自身			GetModuleFileName((HMODULE)hInst,&str[4],sizeof(str)-10);			GetModuleFileName(NULL, szPathOrig, _MAX_PATH);			GetTempPath(_MAX_PATH, szPathClone);			GetTempFileName(szPathClone, __TEXT("Del"), 0, szPathClone);			CopyFile(szPathOrig, szPathClone, FALSE); 			hfile = CreateFile(szPathClone, 0, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_FLAG_DELETE_ON_CLOSE, NULL); 			hProcessOrig = OpenProcess(SYNCHRONIZE, TRUE, GetCurrentProcessId()); 			wsprintf(szCmdLine, __TEXT("%s %d \"%s\""), szPathClone, hProcessOrig, szPathOrig); 			ZeroMemory(&si, sizeof(si));			si.cb = sizeof(si);			CreateProcess(NULL, szCmdLine, NULL, NULL, TRUE, 0, NULL, NULL, &si, π);			CloseHandle(hProcessOrig); 			CloseHandle(hfile);			SendMessage(hwnd,WM_CLOSE,0,0);			break;		case IDC_BUTTON2:                   //弹出About对话框			DialogBox(hInst,"IDD_DIALOG2",hwnd,AboutDlg);			break;		case IDC_RADIO1:                    //按厘米计算			size=100;			break;		case IDC_RADIO2:                    //按分米计算			size=1000;			break;		case IDC_RADIO3:                    //按米计算			size=10000;			break;		case IDC_RADIO4:                    //按百米计算			size=1000000;			break;		case IDC_RADIO5:                    //按千米			size=10000000;			break;		case IDC_CHECK1:                    //启动穿透模式			if(BST_CHECKED==IsDlgButtonChecked(hwnd,IDC_CHECK1))				chuang=TRUE;			if(BST_UNCHECKED==IsDlgButtonChecked(hwnd,IDC_CHECK1))				chuang=FALSE;			break;		}		return TRUE;	case WM_CLOSE:                          //退出程序		ReleaseDC(((HWND)0),hdc);		KillTimer(hwnd,ID_TIMER);		EndDialog(hwnd,LOWORD(wParam));		return TRUE;	default:		return 0;	}	return 0;}

穿透模式，当鼠标向桌面外侧滑动时，会自动从另一侧进入，也是在定时器中检测鼠标位置，实时修改。

删除自身，其实就是把自己拷贝一份到系统临时目录下，拷贝时使用关闭后删除，很多卸载程序应该也是这样删除自身的。


正常情况下要通过调用代码逐步创建UI，但这样不但繁琐，而且易错，基本上没法搞。因此通过资源文件，可以预先设计好程序的UI和每个组件的ID和属性，通过系统提供的API函数一次性加载创建。

Linux下的窗口处理程序也是基于消息驱动的，这种方式的好处是，实现了操作系统和程序之间的解耦，过程是异步的，没有消息处理时就阻塞等待。系统增加新消息时应用程序可以不用例会，因为后边还有系统提供过的默认函数处理，应用程序只处理自身关心的部分，扩展性很强。

现在的软件开发模型，抽象层次越来越高，个人认为不管学什么，如果时间有富余，还是要抽点时间了解些底层的实现方式，这样能更好的掌握所学知识。
