C｜结构和结构指针做为函数参数或返回值

2020-11-14
| 科技

函数的一般形式：

函数类型说明型 函数名(形式参数列表){}

（当然也可以不使用参数）

根据有无返回值，可以分为：

void 函数名(形式参数列表){......}

数据类型 函数名(形式参数列表)

{

......

return语句;

}

return语句返回的数据类型必须与函数名前面声明的数据类型一致；

函数定义时定义形参，函数调用时使用实参，形参相当于变量声明，实参相当于变量赋值，实参可以是常量、变量、表达式、函数，但必须具有确定的值。形参和实参的类型应相同或赋值兼容。

形参变量只有在被调用时才分配内存单元，在调用结束后会释放所分配的内存单元。因此，形参只有在函数内部有效。函数调用结束后不能再使用该形参变量。

函数参数和返回类型可以是标量变量（声明为基本类型的变量），也可以是复合变量（结构体等构造类型的变量）。可以是变量本身，也可以是变量指针（指针传递的结构变量地址值是其首地址。）

//返回结构及结构指针的例子。

#include <stdio.h>

#include <stdlib.h>

struct LIST{

int a,b;

}data[3],*p;

//返回结构值

struct LIST make1(int x, int y)

{

struct LIST temp;

temp.a=x;

temp.b=y;

return temp;

}

//返回结构指针

struct LIST *make2(int x, int y)

{

struct LIST *temp;

temp=(struct LIST *)malloc(sizeof(struct LIST));

temp->a=x;

temp->b=y;

return temp;

}

int main()

{

int i;

data[0]=make1(325,8431);

p=make2(855,56);

data[1]=*p;

data[2]=data[0]; //结构可以整体赋值

for(i=0;i<3;i++)

{

printf("data[%d].a=%d,data[%d].b=%d\n", i,data[i].a,i,data[i].b);

}

system("pause");

return 0;

}

运行结果：

data[0].a=325,data[0].b=8431

data[1].a=855,data[1].b=56

data[2].a=325,data[2].b=8431

//使用结构参数和结构指针参数。

#include <stdio.h>

struct LIST{

int a,b;

}data[3];

struct LIST make(int , int); //返回结构的函数

void disp(struct LIST [ ]); //使用结构参数

void display(struct LIST *); //使用结构指针参数

int main()

{

data[0]=make(133,138);

data[1]=make(185,180);

data[2]=make(136,182);

disp(data); //这里的data以结构参数出现

display(data); //这里的data以结构的首地址出现，即结构指针参数

system("pause");

return 0;

}

struct LIST make(int x, int y)

{

struct LIST temp;

temp.a=x;

temp.b=y;

return temp;

}

void disp(struct LIST data[ ])

{

int i;

for(i=0;i<3;i++)

{

printf("data[%d].a=%d,data[%d].b=%d\n", i,data[i].a,i,data[i].b);

}

}

void display(struct LIST *p)

{

int i;

for(i=0;i<3;i++)

{

printf("data[%d].a=%d,data[%d].b=%d\n", i,(p+i)->a,i, (p+i)->b);

}

}

运行结果：

data[0].a=133,data[0].b=138

data[1].a=185,data[1].b=180

data[2].a=136,data[2].b=182

data[0].a=133,data[0].b=138

data[1].a=185,data[1].b=180

data[2].a=136,data[2].b=182

//结构体作为函数参数

#include <stdio.h>

struct time

{

long int day;

long int hour;

}

main ()

{

struct time minute;

printf ("input day:\n");

scanf ("%ld",&minute.day);

printf ("input hour:\n");

scanf ("%ld",&minute.hour);

print (minute);

system("pause");

}

print (struct time minute)

{

long int i;

i=(minute.day*24+minute.hour)*60;

printf ("total minute is:%ld\n",i);

}

运行结果

input day:

100

input hour:

10

total minute is:144600
