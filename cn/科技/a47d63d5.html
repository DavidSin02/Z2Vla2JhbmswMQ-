MASM 汇编语言——显示彩色字符

编程：在屏幕中间显示80行若干种（可以随时更换，位于color segment）颜色的字符串(可更换位于data segment)，主要学习数据段、指令段和栈的使用，编写简单的calll , ret模块调用。

assume cs:code,ds:data,ss:stack;栈段stack segment dw 0,0,0,0,0,0,0,0,0stack ends;打印的数据段;一般屏幕宽度为160byte，每2byte显示一个字符，其中低位是ASCII码，高位是字符属性;一共可以显示80个字母或数字字符;字符数可以更改data segment  db 'Welcome to Masm!',0 ;以0结尾符  ;db 4ah,71h,4fh,02h,31h,3ah,0 data ends ;字符颜色属性数据段;每一行字符一种颜色属性; 7 6 5 4 3 2 1 0;闪烁 R G B 高亮 R G B; 背景 前景;R:红色 G：绿色 B：蓝色 按位设置属性字节，配出各种不同color segment  db 4ah,71h,4fh,02h,31h,3ah,42h,7ah,7fh,41h,0 ;以0结尾符color ends;代码段code segmentstart: mov ax,data ;data,color,stack段初始化 mov ds,ax mov ax,color mov es,ax mov ax,stack mov ss,ax mov bx,0 mov si,0 mov di,0a0h ;add di,2 mov ax,0 mov cx,0 lp: ;获取字符长度 cmp byte ptr [si],0 je s inc si inc cx jmp lp  s:  mov bx,0 mov ax,cx ;mov si,ax ;mov dl,2 ;div dl mov dx,80 ;计算显示字符开始空格数 sub dx,ax ;mov dh,0 add di,dx call s3 mov bx,0 mov cx,80 ;显示80行数据  s1: cmp bx,dx ;字符颜色种类循环使用 jne s2 mov bx,0   s2: mov ah,byte ptr es:[bx]；获取字符颜色 mov al,0 inc bx push bx push di push cx mov cx,si mov bx,0  s0:  mov dl,ds:[bx] ;显示一行字符 mov dh,ah push ax mov ax,0B800h push ds mov ds,ax mov ds:[di],dx pop ds pop ax inc bx add di,2  loop s0  pop cx pop di add di,0a0h ;add di,0a0h ;pop bx call s3 pop bx  loop s1  ;程序结束，控制权回操作系统 mov ax,4c00h int 21h ;获得字符颜色属性种类  ;get color number ;register (ax) ;retur(dx)   s3: mov bx,0 mov dx,0 s4: cmp byte ptr es:[bx],0 je s5  inc dx inc bx jmp s4  s5: mov bx,0  ret   code ends end start
