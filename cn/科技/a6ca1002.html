Java中Collection集合ArrayList和HashSet的使用

2020-11-14
| 科技
package dao;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashSet;
/*
* 本类讲解java中的Collection集合的使用
*/
public class CollectionDemo {
public static void main(String[] args) {
//要使用Collection集合 第一步需要创建Collection的对象
//ArrayList是Collection的一个实现类
//Collection有很多个实现了 ArrayList是其中的一种
//要使用以下的两个类得先导入包
System.out.println("-------以下是用ArrayList子类来创建集合对象------");
Collection demo1 = new ArrayList();
//对象创建好了 接下来就是往对象里面放东西了
//注意 ：集合中可以存放任意类型的数据，因为Collection会把这些统一识别为object对象
demo1.add("北京");//用提供的add方法像集合中添加内容
demo1.add("上海");
demo1.add(100);
demo1.add("天津");
//获取集合中的个数
System.out.println("demo1的元素个数为："+demo1.size());//输出集合中的个数
//输出集合中的所有值
//这个地方可以看到输出的值为一个连续的值，吧集合中所有的值都输出了
//原因是 因为 Collection集合的所有子类都重写了toString方法。该方法可以一次性的输出集合中的所有元素
System.out.println("移除天津之前的值为："+demo1);
//移除某个指定的元素 对象名.remove(元素值)
demo1.remove("天津");
//输出移除之后的
System.out.println("移除天津之后的值为："+demo1);
//判断集合中是否存在某个指定的元素 contains
System.out.println("demo1集合中是否存在上海字符串："+demo1.contains("上海"));
System.out.println("demo1集合中是否存在广州字符串："+demo1.contains("广州"));
//一次性删除demo1集合中的所有元素
demo1.clear();
//clear之后的值为空值 因为集合被clear之后就清空了所有的元素
System.out.println("clear demo1集合之后的值为："+demo1);
System.out.println("\n-------以下是用HashSet子类来创建集合对象------");
//创建第二个Collection对象 用子类HashSet实现
//同样也需要导入HashSet的包
Collection demo2 = new HashSet();
demo2.add("海淀区");
demo2.add("朝阳区");
demo2.add("东城区");
demo2.add("西城区");
//输出
System.out.println("移除海淀区之前的值为:"+demo2);
demo2.remove("海淀区");
System.out.println("移除海淀区之后的值为:"+demo2);
//判断一个集合是否完全包含另一个集合 结果为false
System.out.println("demo1集合是否包含demo2集合:"+demo1.containsAll(demo2));
//小结：在该案例里面主要讲解了Collection接口用它的两个实现类来创建对象
// 虽然demo1和demo2是两个不同的实现类创建的对象
// 但是却拥有相同的方法 ，因为它们都同时是Conllction的子类
// 所以在使用add size remove clear contains时没有任何的区别
/*
*注意：当在运行以上代码的时候编译器会提示黄色波浪线的警告，这个是在提醒用户没有使用泛型
* 来限制集合的元素类型，这个问题需要用到后面的泛型集合来解决 这个章节里无须理会
* 普通情况下集合都会把它的元素定义为对象类型，所以无论什么类型都可以直接扔进集合中
*/
}
}
