二、获取影评数据并写入文件

commentList = []for i in range(0, 501, 20):    url = 'https://movie.douban.com/subject/26357307/comments?start='+str(i)+'&limit=20&sort=new_score&status=P'    driver.get(url)    source = driver.page_source    pattern = re.compile('<span class="short">(.*?)</span>')    comments = re.findall(pattern, source)    commentList.extend(comments)    time.sleep(5)with open('shortComments.txt', 'w', encoding='utf-8') as f:    for comment in commentList:        f.write(comment + '\n')
