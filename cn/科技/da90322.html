SpringBoot 集成StateMachine实现状态机

2020-10-29
| 科技

前言：

Spring StateMachine框架是Spring.io提供给开发者一种可以简化状态机实现的过程，让状态机结构更加层次化。本次状态机的实现Showcase主要是模拟一个商品购买的场景，从下单到收货这个业务流程下所涉及的事件、状态来描述Spring StateMachine如何使用与运作。

正文：

下面详细的介绍整个实现过程：

States.java完整代码如下：

package com.example.statemachine.enums;/** * @package: com.example.statemachine.enums * @description: 状态枚举类 * @author: @IT讲坛 * @create: 2018-12-16 17:27 **/public enum States { MAKE_AN_ORDER, // 生成订单 UNPAID, // 待支付 PAYING, // 支付中 PAYED, // 已支付 WAITING_FOR_RECEIVE, // 待收货 CUSTOMER_RECEIVED, // 已收货 DONE // 订单结束}

3.创建StateMachine的配置类，这个配置类也是Spring Boot的自动装配的必备，也是大家从xml配置转为bean配置的最不适应的地方

StateMachineConfig.java完整代码如下：

StateMachineConfig

package com.example.statemachine.config;import com.example.statemachine.enums.Events;import com.example.statemachine.enums.States;import com.example.statemachine.persist.InMemoryStateMachinePersist;import org.springframework.beans.factory.annotation.Autowired;import org.springframework.context.annotation.Bean;import org.springframework.context.annotation.Configuration;import org.springframework.statemachine.config.EnableStateMachine;import org.springframework.statemachine.config.EnumStateMachineConfigurerAdapter;import org.springframework.statemachine.config.builders.StateMachineStateConfigurer;import org.springframework.statemachine.config.builders.StateMachineTransitionConfigurer;import org.springframework.statemachine.persist.DefaultStateMachinePersister;import org.springframework.statemachine.persist.StateMachinePersister;import java.util.EnumSet;/** * @package: com.example.statemachine.config * @description: 状态机配置类 * @author: @IT讲坛 * @create: 2018-12-16 17:17 **/@Configuration@EnableStateMachine(name="stateMachine") //可以通过设置名称来定义若干个状态机的Config，并可以通过制定状态机配置名称来使用制定的状态机 @EnableStateMachine(name="testMachine")public class StateMachineConfig extends EnumStateMachineConfigurerAdapter<States, Events> { @Autowired private InMemoryStateMachinePersist inMemoryStateMachinePersist; @Bean(name = "myPersister") public StateMachinePersister<States, Events, String> getPersister() { return new DefaultStateMachinePersister<>(inMemoryStateMachinePersist); } /** * * @title 初始化状态机及初始状态 * * @description * * @see org.springframework.statemachine.config.AbstractStateMachineConfigurerAdapter#configure(org.springframework.statemachine.config.builders.StateMachineStateConfigurer) * @param states * @throws Exception */ @Override public void configure(StateMachineStateConfigurer<States, Events> states) throws Exception { states .withStates() .initial(States.MAKE_AN_ORDER) //初始化状态为UNPAID .states(EnumSet.allOf(States.class)); } /** * * @title 事件及状态流转配置 * * @description * * @see org.springframework.statemachine.config.AbstractStateMachineConfigurerAdapter#configure(org.springframework.statemachine.config.builders.StateMachineTransitionConfigurer) * @param transitions * @throws Exception */ @Override public void configure(StateMachineTransitionConfigurer<States, Events> transitions) throws Exception { transitions .withExternal() .event(Events.ORDERING) //当事件为ORDERING时，源状态从MAKE_AN_ORDER变为UNPAID .source(States.MAKE_AN_ORDER).target(States.UNPAID) .and() .withExternal() .event(Events.PAY) //当事件为PAY时，源状态从UNPAID变为PAYING .source(States.UNPAID).target(States.PAYING) .and() .withExternal() .event(Events.PAYED) //当事件为PAYED时，源状态从PAYING变为PAYED .source(States.PAYING).target(States.PAYED) .and() .withExternal() .event(Events.RECEIVE) //当事件为RECEIVE时，源状态从PAYED变为WAITING_FOR_RECEIVE .source(States.PAYED).target(States.WAITING_FOR_RECEIVE) .and() .withExternal() .event(Events.RECEIVED) //当事件为RECEIVED时，源状态从WAITING_FOR_RECEIVE变为CUSTOMER_RECEIVED .source(States.WAITING_FOR_RECEIVE).target(States.CUSTOMER_RECEIVED) .and() .withExternal() .event(Events.FINISH) //当事件为FINISH时，源状态从CUSTOMER_RECEIVED变为DONE .source(States.CUSTOMER_RECEIVED).target(States.DONE); }}

说明：

1)注解：@Configuration和@EnableStateMachine，一个是表明此类为一个Spring的Bean配置类，另一个则是表明当前的配置是要启用状态机的自动装配。

2)父类EnumStateMachineConfigurerAdapter<States,Events>，为状态机的自动装配提供自定义的配置重写，需要重写configure方法。

3)可以暂时忽略本类中的关于状态持久化的配置，因为，这个配不配不影响一个状态机Showcase的实现，例如：InMemoryStateMachinePersist

4)configure(StateMachineStateConfigurer<States, Events> states)throws Exception 方法是用来定义状态机的初始化状态以及所能包含的状态集合。

5)configure(StateMachineTransitionConfigurer<States, Events> transitions)throws Exception 方法是用来定义状态机由哪些事件导致状态从哪个状态迁移到哪个状态，其中命名中我们很容易理解每一个迁移动作，source是源状态，target是目标状态，event为触发事件。

4.配置事件处理者，用于监听状态变化，允许当状态发生变化时则执行对应的方法来进行相关业务处理

EventProcessor.java:

EventProcessor

package com.example.statemachine.handle;import lombok.extern.slf4j.Slf4j;import org.springframework.statemachine.annotation.OnTransition;import org.springframework.statemachine.annotation.WithStateMachine;/** * @package: com.example.statemachine.handle * @description: 事件处理类 * @author: @IT讲坛 * @create: 2018-12-17 09:17 **/@WithStateMachine@Slf4jpublic class EventProcessor{ /** * * @title 自定义事件执行方法——1.创建订单 * * @description 当状态为MAKE_AN_ORDER，并转为UNPAID状态就执行此方法 * */ @OnTransition(source="MAKE_AN_ORDER",target = "UNPAID") public void makeAnOrder() { log.info("订单已创建，等待用户支付"); } /** * * @title 自定义事件执行方法——6.完成订单 * * @description 当状态为RECEIVED，并转为DONE状态就执行此方法 * */ @OnTransition(source = "CUSTOMER_RECEIVED", target = "DONE") public void finishOrder() { log.info("订单流程已完成"); }}

EventProcessor2.java

EventProcessor2

package com.example.statemachine.handle;import lombok.extern.slf4j.Slf4j;import org.springframework.statemachine.annotation.OnTransition;import org.springframework.statemachine.annotation.OnTransitionEnd;import org.springframework.statemachine.annotation.OnTransitionStart;import org.springframework.statemachine.annotation.WithStateMachine;/** * @package: com.example.statemachine.handle * @description: 事件处理类 * @author: @IT讲坛 * @create: 2018-12-17 09:17 **/@WithStateMachine@Slf4jpublic class EventProcessor2 { /** * * @title 自定义事件执行方法——2.开始支付 * * @description 当状态为UNPAID，并转为PAYING状态就执行此方法 * */ @OnTransitionStart(source = "UNPAID", target = "PAYING") public void paying() { log.info("用户支付中....."); } /** * * @title 自定义事件执行方法——3.完成支付 * * @description 当状态为PAYING，并转为PAYED状态就执行此方法 * */ @OnTransition(source = "PAYING", target = "PAYED") public void payed() { log.info("用户完成支付"); } /** * * @title 自定义事件执行方法——并行 4.1.收到支付成功短信 * * @description 当状态为PAYED，并转为WAITING_FOR_RECEIVE状态就执行此方法 * */ @OnTransitionEnd(source = "PAYED", target = "WAITING_FOR_RECEIVE") public void syncNotifyFromSMS() { log.info("收到用户支付成功短信，准备发货"); } /** * * @title 自定义事件执行方法——并行 4.2.发货中 * * @description 当状态为PAYED，并转为WAITING_FOR_RECEIVE状态就执行此方法 * */ @OnTransitionEnd(source = "PAYED", target = "WAITING_FOR_RECEIVE") public void waitingForReceive() { log.info("等待用户签收货品......"); } /** * * @title 自定义事件执行方法——5.用户完成收货 * * @description 当状态为WAITING_FOR_RECEIVE，并转为RECEIVED状态就执行此方法 * */ @OnTransitionEnd(source = "WAITING_FOR_RECEIVE", target = "CUSTOMER_RECEIVED") public void received() { log.info("用户已签收货品"); }}

说明:

其实一个EventProcessor类就可以完成需要的处理，但为什么要这里要分成两个Processor类来实现，主要原因就是想简单设计一下对于处理过程的抽象，把开始与结束放在一个类中，具体过程放在另外一个类中；此类有几点需要注意：

1)注解@WithStateMachine，定义此类为一个状态监听者，@OnTransition(source="",target = "")，主要是定义一个方法的执行时机，也就是当源状态是A，目标状态是B时，则执行这个方法

2) ILog接口，这个接口是自己为了方便打印日志，自己封装的一个Logback logger的Handler，对于状态机的showcase没有影响，各位可以直接忽略。

5.最后创建主入口类来完成整个Showcase的流程

StateMachineApplication.java

StateMachineApplication

package com.example.statemachine;import com.example.statemachine.enums.Events;import com.example.statemachine.enums.States;import lombok.extern.slf4j.Slf4j;import org.springframework.beans.factory.annotation.Autowired;import org.springframework.beans.factory.annotation.Value;import org.springframework.boot.CommandLineRunner;import org.springframework.boot.SpringApplication;import org.springframework.boot.autoconfigure.SpringBootApplication;import org.springframework.statemachine.StateMachine;import org.springframework.statemachine.persist.StateMachinePersister;import javax.annotation.Resource;@SpringBootApplication@Slf4jpublic class StatemachineApplication implements CommandLineRunner { @Value("${spring.application.name}") private String proName; /** 状态机对象 */ @Resource //使用制定的状态机配置进行注入，例如：@Resource(name="testMachine"); 这样可以有多个状态机实例 private StateMachine<States, Events> stateMachine; /** 状态机持久化 */ @Autowired private StateMachinePersister<States, Events, String> persist; public static void main(String[] args) { SpringApplication.run(StatemachineApplication.class, args); } /** * * @title 事件驱动业务流程主控方法 * * @description * 1.启动状态机 * 2.发送事件 * 3.将某一事件进行持久化 * 4.打印状态机的当前状态 * 5.取回被持久化的状态并打印  * * * @see org.springframework.boot.CommandLineRunner#run(java.lang.String[]) * @param args * @throws Exception */ @Override public void run(String... args) throws Exception { log.info("==========================================Project name : {}" , proName); stateMachine.start(); stateMachine.sendEvent(Events.ORDERING); stateMachine.sendEvent(Events.PAY); stateMachine.sendEvent(Events.PAYED); persist.persist(stateMachine, "MyRegion"); log.info("Store the state in MyRegion ================1: {}",stateMachine.getState().getId()); stateMachine.sendEvent(Events.RECEIVE); stateMachine.sendEvent(Events.RECEIVED); stateMachine.sendEvent(Events.FINISH); log.info("The state Now ================2: {}",stateMachine.getState().getId()); persist.restore(stateMachine, "MyRegion"); log.info("Restore the state from MyRegion ================3: {}",stateMachine.getState().getId()); }}

说明：

在run方法中，定义了流程的驱动处理过程，其中start()是根据之前的定义触发状态机启动这个订单流程，并一步步的sendEvent来持续驱动，直至最终的FINISH事件发出。

注意几点：

1)persist.persist和persist.restore这两个方法调用是关于持久化和获取这个流程状态的，可以不用关心，除非想了解如何持久化状态机内瞬时状态及状态获取的。

2)直接运行这个Main函数、JUnit或使用maven客户端来执行spring boot cli命令，例如： clean compile spring-boot:run

至此，就完成了一个模拟订单下单场景的状态机驱动过程的Showcase。
