自定義菜單欄

Application.CommandBars.AddName:="菜單名稱",Temporary:=True

Application.CommandBars("菜單名稱").Visible=True

Application.CommandBars("菜單名稱").Delete

For Each bar In Application.CommandBars

If bar.BuiltIn=FalseThen

MsgBox本工作簿創建了自定義工具欄，其名稱為：&bar.Name

End If

Next

設置菜單命令

With Application.CommandBars(x).Controls.Add(Type:=msoControlButton,Before:=y)

.Caption="命令名稱"

.OnAction="宏名"

End With

x為：

data

tools

cell

Application.CommandBars(x).Controls(y).Enabled=False/True

Application.CommandBars(x).Controls(y).Delete

Application.CommandBars(x).Controls(y).Reset


設置菜單命令控件

With Application.CommandBars(x).Controls(y).Controls(z)

.BeginGroup=True/False

.Execute

.Enabled=False/True

.Delete

.Reset

End With


設置子菜單

Application.CommandBars(x).Controls(y).Controls.Add(Type:=msoControlPopup,Before:=z).Caption="命令名稱"

With CommandBars(x).Controls(y).Controls(z)

.Controls.Add(Type:=msoControlButton,Before:=1).Caption="命令名稱"

.Controls("命令名稱").OnAction="宏名"

End With

With Application.CommandBars(x).Controls(y).Controls(z).Controls(z1)

.Enabled=False/True

.Delete

.Reset

End With
