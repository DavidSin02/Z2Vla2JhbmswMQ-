操作系統原理實驗—進程調度實驗報告

一、目的與要求

（1）進程是操作系統最重要的概念之一，進程調度是操作系統內核的重要功能，本實驗選用XX語言編寫了一個進程調度模擬程序，使用優先級或時間的輪轉法實現進程調度。本實驗可加深對進程調度算法的理解。

（2）按照實驗題目要求獨立地、正確地完成實驗內容（編寫、調試算法程序，提交程序清單及相關實驗數據與運行結果，完成個人實驗報告）。

（3）2020年6月5日以前提交本次實驗報告（含實驗所有電子文檔和紙質報告，實驗相關文件打包以 學號-姓名-實驗1.rar 為名命名，實驗報名以 學號-姓名-實驗1為文件名，由學習委員以班為單位統一打包提交）。

2 實驗內容或題目

（1）設計有5個或5個以上進程併發執行的模擬調度程序，每個程序由一個PCB表示。

（2）模擬調度程序可任選兩種調度算法之一實現，也可實驗兩個或兩個以上調度算法。

（3）程序執行中應能在屏幕上顯示出各進程的狀態變化，以便於觀察調度的整個過程。

（4）本次實驗內容（項目）的詳細說明以及要求請參見實驗指導書。


3 實驗步驟與源程序

1. 算法原理

時間片輪轉調度算法

1) 時間片輪轉算法的思想是，讓隊列中運行的每個進程都可以得到平等的運行時間片，然後對隊列進行循環直到所有進程執行完畢。

2) 對於時間片大小的選取是十分重要的，既不能太大，如果太大的話會導致短進程無法作業運行，如果時間片太小的話效率又會減小很多。所以對於時間片來講要根據進程的大小選擇合理的時間片

3) 進程的切換機制體現出了時間片輪轉算法的特點。若一個進程在時間片還沒結束時就已完成，就將它從隊列中移除。若一個進程在時間片結束時還沒運行完成，等待下次運行。

4) 本次實驗的時間片輪轉算法採用c語言對算法進行模擬，採用的時間片為3直到系統需要的時間片都小於0結束

優先級調度算法

1） 由於時間的輪轉算法，無法對進程的緊急情況進行區分。因此又提出了優先級調度算法，從而解決進程優先級的問題。

2） 進程優先級的確定同樣重要，進程優先級可以分為靜態優先級和動態優先部，靜優先級是在通程創建裡初期就被確定的值，此後不再更改。動態優先級指進程在創建時被賦子一個初值，此後其值會所進程的推進或等待時間的增加而改變。

3） 本次實驗的優先級調度算法採用C語言對算法進行模擬，優先級選擇初始化隨機數據1-5，當need滿足的時候將進程優先級設置為-100。

/**

時間的輪轉算法！！！

#include <stdio.h>

#include <conio.h>

#include <stdlib.h>

#include <time.h>


#define spand 3//設置時間片長度為3


typedef struct pcd

{

int id;//進程ID

int span;//運行進程的時間片數

int used;//已用的時間片數

int need;//進程需要運行的時間

char status;//進程狀態

struct PCB *next;

} PCB;


PCB *head,p[5],t[5];


//打印

void print()

{

int i,temp;

printf("---------------------------------\n");

printf("ID \t span\tused\tneed\tstatus\n");


for(i=0; i < 5; i++)

{

printf("%d\t%d\t%d\t%d\t%c\n",p[i].id,p[i].span,p[i].used,p[i].need,p[i].status);

}


}


//初始化進程隊列

void inputprocess()

{

int i;

srand(time(0));//為隨機數產生種子；


for(i = 0; i < 5; i++)

{

p[i].id = i + 1;

p[i].span= 3;

p[i].used=0;

p[i].need=5+rand()%6;//為總的時間片產生5-10的整數

p[i].status='J';

}


head = p;

printf("開始進程的狀態：\n");

print();//打印進程狀態


}


//時間的輪轉算法

void runprocess()

{


int i=0;


while(p[0].need > 0 ||p[1].need > 0||p[2].need > 0||p[3].need > 0||p[4].need > 0)//當需求有一個未被滿足時進程繼續執行

{


p[i%5].used += 3;

p[i%5].need -= 3;

p[i%5].status = 'R';

if(p[i%5].need <= 0)//如果進程已被滿足調用下一個進程

{

p[i%5].status='F';

i++;


}

else if(p[i%5].need <= 0 && p[(i%5)+1].need <= 0)//如果連續兩個進程已被滿足調用下一個進程

{

p[i%5].status='F';

i++;

}

else if(p[i%5].need <= 0 && p[(i%5)+1].need <= 0 && p[(i%5)+2].need <= 0)//如果連續三個進程已被滿足調用下一個進程

{

p[i%5].status='F';

i++;

}

else if(p[i%5].need <= 0 && p[(i%5)+1].need <= 0 && p[(i%5)+2].need <= 0 && p[(i%5)+3].need <= 0)////如果連續三個進程已被滿足調用下一個進程

{

p[i%5].status='F';

i++;

}

print();//打印進程狀態

if(p[i%5].status=='R')//如果本次進程執行完畢將程序由運行狀態置為就緒狀態

{

p[i%5].status = 'J';

}

i++;


}


}


int main(void)

{

inputprocess();

runprocess();


return 0;

}


/**

優先級調度算法

*/


#include <stdio.h>

#include <conio.h>

#include <stdlib.h>


typedef struct pcd

{

int id;//進程ID

int prior;//進程優先級

int used;//已使用時間

int need;//還需時間

char status;//進程狀態

} PCB;


PCB *head,p[5],a[5];


//將每次進程調度的結果打印在屏幕上

void print()

{

int i,temp;

printf("---------------------------------\n");

printf("ID \t prior\tused\tneed\tstatus\n");


for(i=0; i < 5; i++)

{

printf("%d\t%d\t%d\t%d\t%c\n",p[i].id,p[i].prior,p[i].used,p[i].need,p[i].status);

}


}


//根據進程的優先級進行排序為下一次進程調度做準備

void sort()

{

int i,j;

//PCB temp;


for(i=0; i < 5; i++)

{

for(j = 0; j <4-i;j++)

{

if(p[j].prior <p[j+1].prior)

{

PCB temp = p[j];

p[j] = p[j+1];

p[j+1] = temp;


}


}

}

}


//初始化進程隊列

void inputprocess()

{

int i;


for(i = 0; i < 5; i++)

{

p[i].id = i + 1;

p[i].prior = rand()%5+1;//隨機生成1-5的整數

p[i].used = 0;

p[i].need = 5+rand()%6;//為總的時間片產生5-10的整數

p[i].status = 'W';


}


sort();//按優先級排序

head = p;

printf("開始進程的狀態：\n");

print();//打印進程狀態


}


//按最高優先級調度進程隊列

void runprocess()

{

PCB *p1 = NULL;


do{


p1 = head;

head = p+1;


printf("***%d***%d***\n",p1->prior,head->prior);


while((p1->need > 0)&&(p1->prior >= head->prior))//如果p1的需求數大於0並且 當前進程的優先級大於等於它下面一個進程的優先級

{

p1->need--;

p1->prior--;

p1->used++;

p1->status='R';

print();

}


if(p1->need == 0)//如果當前進程的需求為0時將優先級置為-100說明該進程運行完畢

{

p1->prior = -100;

p1->status = 'F';

}

else//否則將該進程置為等待狀態為下一次運行做準備

{

p1->status='W';

}

sort();

head = p;

}

while(head->prior != -100);

print();//顯示最後各個進程的狀態


printf("--------------------------\n");

printf("所有進程全部完成！！！");

}


int main()

{

inputprocess();//調用初始化進程自定義函數

runprocess();//調用調度程序的函數

return(1);

}

6.分析與思考

1、邏輯時間片該如何實現?

答:系統將所有的就緒進程按先來先服務算法的原則，排成- 一個隊列，每次調度時，系隊列首進程，並讓其執行一一個時間片。當執行的時間片用完時，由一個計時器發出時鐘中斷請求，調度程序根據這個請求停止該進程的運行，將它送到就緒隊列的末尾，再把處理機分給就緒隊列中

新的隊首進程，同時讓它也執行一一個時間片。

2、如果不使用指針操作，是否也可以使用數組實現進程就緒隊列的組織?

答:可以
