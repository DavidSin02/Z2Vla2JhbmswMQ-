第八章：Python+matplotlib座標軸的高階應用

2020-10-29
| 科技

1、向畫布當中的任意位置添加任意數量的座標軸

 1 import  matplotlib.pyplot as plt 2 import numpy as np 3  4 # 左、下分別表示座標軸的左側邊緣和底部邊緣距離畫布的距離 5 # 寬、高分別表示座標軸的寬度和高度 6 plt.axes([0.05,0.7,.3,.3],      # 左，下，寬，高 7          frameon=True,          # True表示繪製座標軸的四條軸脊,也就是四條邊框 8          aspect="equal")        # 控制圖像的寬高比例 9 plt.plot(np.arange(3),          # X座標軸刻度10          [0,1,0],               # 曲線值11          color="blue",          # 曲線的顏色12          linewidth=2,           # 曲線寬度13          linestyle="--")        # 曲線樣式14 15 plt.axes([0.3,0.4,.3,.3],16          frameon=True,17          aspect="equal")18 plt.plot(2+np.arange(3),19          [0,1,0],20          color="red",21          linewidth=2,22          linestyle="-")23 24 plt.axes([0.55,0.1,.3,.3],25          frameon=True,26          aspect="equal")27 plt.plot(4+np.arange(3),28          [0,1,0],29          color="c",30          linewidth=2,31          linestyle=":")32 33 plt.show()

3、使用兩種方法控制座標軸的刻度顯示

（1） 使用set_xticks和set_xticks控制

 1 import  matplotlib.pyplot as plt 2  3 ax1 = plt.subplot(121) 4 ax1.set_xticks(range(0,251,50)) 5 ax1.set_yticks(range(0,251,50)) 6 plt.grid(True,axis="x") 7  8 ax2 = plt.subplot(122) 9 ax2.set_xticks([])10 ax2.set_yticks(range(0,5,1))11 plt.grid(True,axis="x")12 13 plt.show()

（2） 調用函數setp()

 1 import  matplotlib.pyplot as plt 2  3 ax1 = plt.subplot(221) 4 # 若要設置Y軸的刻度和刻度線則把get_xticklabels()改成get_yticklabels() 5 plt.setp(ax1.get_xticklabels(),     # 設置X軸刻度標籤 6          visible=True)              # X軸刻度標籤可顯示 7 plt.setp(ax1.get_xticklines(),      # 設置X軸刻度線 8          visible=True)              # X軸刻度線可顯示 9 plt.grid(True,axis="x")10 11 ax2 = plt.subplot(222)12 plt.setp(ax2.get_xticklabels(),13          visible=True)              # X軸刻度標籤可顯示14 plt.setp(ax2.get_xticklines(),15          visible=False)             # X軸刻度線不可顯示16 plt.grid(True,axis="x")17 18 ax3 = plt.subplot(223)19 plt.setp(ax3.get_xticklabels(),20          visible=False)             # X軸刻度標籤不可顯示21 plt.setp(ax3.get_xticklines(),22          visible=True)              # X軸刻度線可顯示23 plt.grid(True,axis="x")24 25 ax4 = plt.subplot(224)26 plt.setp(ax4.get_xticklabels(),27          visible=False)              # X軸刻度標籤不可顯示28 plt.setp(ax4.get_xticklines(),29          visible=False)              # X軸刻度線不可顯示30 plt.grid(True,axis="x")31 32 plt.show()

4、棉棒圖的定製化展示

 1 import  matplotlib.pyplot as plt 2 import numpy as np 3  4 x = np.linspace(0.5,2*np.pi,20) 5 y = np.random.randn(20) 6  7 # 獲得實例，前兩個實例的屬性值的改變用setp,stemlines是實例列表 8 markerline,stemlines,baseline = plt.stem(x,y) 9 10 plt.setp(markerline,        # X,Y的組合點，如（1,3）11          color="c",         # 數據點的顏色12          marker="D")        # 數據點的表示類型13 plt.setp(stemlines,         # 數據點到基線（紅色橫線）的實例14          linestyle="-.")    # 樣式15 baseline.set_linewidth(20)  # 基線實例（紅色的橫線）並設置寬度16 17 plt.show()
