void trimSpace(char *str)

{

int i, j, k;

char tmpstr[MAXSTRSIZE]; // 此處用戶可動態分配，滿足情變

i = strlen(str);

for (k = j = 0; j<i; j++)

{

if (str[j] != ' ')

{

tmpstr[k] = str[j]; k++;

}

}

tmpstr[k] = '\0';

strcpy(str, tmpstr);

}
