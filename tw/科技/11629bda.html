main()

{

if (x>y)

{t=x;x=y;y=t;} /*交換x,y的值*/

if(x>z)

{t=z;z=x;x=t;}/*交換x,z的值*/

if(y>z)

{t=y;y=z;z=t;}/*交換z,y的值*/

printf("small to big: %d %d %d\n",x,y,z);

}
