8421編碼開關

8421編碼開關種類繁多，按操作方式分有指拔式、旋轉式；按制式分有二進制、十進制、十六進制等。每一位都有一個獨立的BCD編碼撥動盤，我們每選擇的一個數字都由裡面的印刷電路產生BCD編碼，分別由下面的4個管腳輸出。組合非常靈活，我們可以根據設計的需要，任意組合多個BCD編碼撥動盤來組成多位的應用。

在單片機設計中，特別是工業應用的設計中，一些外部的數值設置我們經常會選擇用到它。使用起來也比較方便。

一個8421編碼拔動盤，如果是十進制的是數字0-9，如果是十六進制的數字是0-F。也就是說十進制有10個數字，十六進制有十六個數字。而8421編碼正好可以編碼16個數字，它的編碼方式參考圖1。

讀取編碼盤的工作原理

單片機上電覆位時，W1-W4保持高電平。我們用它們來讀取各數據位的狀態，在工作期間不去變更它們的電平狀態，它們的電平狀態將由D0-D3來決定。

在讀取數據時，首先把D0-D4全部拉到高電平。D0拉為低電平，延時一段時間再分別去讀取W1-W4的狀態就可以知道4位編碼盤的D0是什麼狀態。W1-W4讀到的是低電平，那麼說明相應4位編碼盤的D0被編碼了。用相同的方式去操作D1-D3就可以把完整的數值讀取出來。

下面用程序來說明這一操作過程

/***********************拔碼開關應用程序*************************/

#include <STC15F2K60S2.H>

int t_1ms, t_50ms; //這裡定義的是1ms和50ms定時器，詳細說明在上一課中有詳細介紹

unsigned int Read_Switch[2], Read_Switch_Dec[2]; //定義讀取到編碼開關的16進制和10進制值的存儲數組

void Read_8421_Switch();

void main()

{

Read_8421_Switch();

}

void Read_8421_Switch() //指拔開關設置值讀取

{

static float scantime = 0;

static bit reachtime = 0, savetime = 0, R_W = 0;

static char switch_bite = 1;

char turnchar = 0, turncnt = 1, i;

static int turnchartmp[1], Read_Switch[1];

int turnmul = 0;

if(!savetime)

{

scantime = t_1ms;

savetime = 1;

}

if((t_1ms - scantime) < 0) //防止定時寄存器溢出時出現負數的情況

scantime = t_1ms;

if(savetime && t_1ms - scantime > 5) //5ms等待延時

reachtime = 1;

if(!reachtime)

goto Read__out;

//這一段設置了一個5ms的狀態保持時間，在時間到了之後才會去讀取W1-W4的狀態

switch(switch_bite) //選擇去拉低D0-D3中的哪一個

{

case 1: //操作D0

if(R_W) //決定是去拉低D0的電平，還是去讀取W1的狀態

{

//讀取D0狀態並寫入暫存寄存器

if(P33)

Read_Switch[0] &= 0xEFFF;

else

Read_Switch[0] |= 0x1000;

if(P35)

Read_Switch[0] &= 0xFEFF;

else

Read_Switch[0] |= 0x100;

if(P36)

Read_Switch[0] &= 0xFFEF;

else

Read_Switch[0] |= 0x10;

if(P37)

Read_Switch[0] &= 0xFFFE;

else

Read_Switch[0] |= 0x1;

P15 = 1; //讀取完成時重新拉高D0，以備下一次的讀取

}else{

P15 = 0; //拉低D0

}

break;

case 2: //操作D1

if(R_W) //決定是去拉低D1的電平，還是去讀取W2的狀態

{

//讀取D1狀態並寫入暫存寄存器

if(P33)

Read_Switch[0] &= 0xDFFF;

else

Read_Switch[0] |= 0x2000;

if(P35)

Read_Switch[0] &= 0xFDFF;

else

Read_Switch[0] |= 0x200;

if(P36)

Read_Switch[0] &= 0xFFDF;

else

Read_Switch[0] |= 0x20;

if(P37)

Read_Switch[0] &= 0xFFFD;

else

Read_Switch[0] |= 0x2;

P16 = 1; //讀取完成時重新拉高D1，以備下一次的讀取

}else{

P16 = 0; //拉低D1

}

break;

case 3: //操作D2

if(R_W) //決定是去拉低D2的電平，還是去讀取W3的狀態

{

//讀取D2狀態並寫入暫存寄存器

if(P33)

Read_Switch[0] &= 0xBFFF;

else

Read_Switch[0] |= 0x4000;

if(P35)

Read_Switch[0] &= 0xFBFF;

else

Read_Switch[0] |= 0x400;

if(P36)

Read_Switch[0] &= 0xFFBF;

else

Read_Switch[0] |= 0x40;

if(P37)

Read_Switch[0] &= 0xFFFB;

else

Read_Switch[0] |= 0x4;

P17 = 1; //讀取完成時重新拉高D2，以備下一次的讀取

}else{

P17 = 0; //拉低D2

}

break;

case 4: //操作D3

if(R_W) //決定是去拉低D3的電平，還是去讀取W4的狀態

{

//讀取D3狀態並寫入暫存寄存器

if(P33)

Read_Switch[0] &= 0x7FFF;

else

Read_Switch[0] |= 0x8000;

if(P35)

Read_Switch[0] &= 0xF7FF;

else

Read_Switch[0] |= 0x800;

if(P36)

Read_Switch[0] &= 0xFF7F;

else

Read_Switch[0] |= 0x80;

if(P37)

Read_Switch[0] &= 0xFFF7;

else

Read_Switch[0] |= 0x8;

P32 = 1; //讀取完成時重新拉高D3，以備下一次的讀取

}else{

P32 = 0; //拉低D3

}

break;

}

savetime = 0; //5ms定時時間到後的操作完成，等待下一次5ms定時

reachtime = 0; //重置定時標誌

R_W = ~R_W; //拉低D0-D3或讀取W1-W4標誌轉換

if(!R_W)

switch_bite++; //D0-D3循環

if(switch_bite > 4) //D0-D3全部讀取完成進行數據處理，把十六進制值轉換為我們設置的十進制值

{

switch_bite = 1;

Read_Switch_Dec[0] = 0;

turnchartmp[0] = Read_Switch[0];

for(i = 0; i < 4; i++)

{

switch(i)

{

case 0:

turnmul = 1;

break;

case 1:

turnmul = 10;

break;

case 2:

turnmul = 100;

break;

case 3:

turnmul = 1000;

break;

}

turnchar = turnchartmp[0] & 0xF;

Read_Switch_Dec[0] = Read_Switch_Dec[0] + turnchar * turnmul;

turncnt = (i + 1) * 4;

turnchartmp[0] = Read_Switch[0] >> turncnt;

}

turnchartmp[0] = Read_Switch[0];

}

Read__out:

_nop_();

}
