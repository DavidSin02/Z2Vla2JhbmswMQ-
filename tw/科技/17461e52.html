分享一個小知識點，二叉樹的幾種遍歷方法(遞歸實現)

2020-11-14
| 科技

就不做多餘的講解相關的原理的,直接上代碼,代碼中有相關的註釋,代碼也比較簡單容易看懂)
結點的代碼:

public class TreeNode {    /**     * 左孩子     */    private TreeNode leftChildren;    /**     * 數據域     */    private Integer data;    /**     * 右孩子     */    private TreeNode rightChildren;    public TreeNode() {    }    public TreeNode(TreeNode leftChildren, Integer data, TreeNode rightChildren) {        this.leftChildren = leftChildren;        this.data = data;        this.rightChildren = rightChildren;    }    public TreeNode getLeftChildren() {        return leftChildren;    }    public void setLeftChildren(TreeNode leftChildren) {        this.leftChildren = leftChildren;    }    public Integer getData() {        return data;    }    public void setData(Integer data) {        this.data = data;    }    public TreeNode getRightChildren() {        return rightChildren;    }    public void setRightChildren(TreeNode rightChildren) {        this.rightChildren = rightChildren;    }}

實現三種排序的代碼,都是可以直接使用的.

public class SortBTree {    public static void main(String[] args) {        /**         * 初始化二叉樹         *                1         *             2     3         *           4   5  6  7         */        TreeNode node7 = new TreeNode(null, 7, null);        TreeNode node6 = new TreeNode(null, 6, null);        TreeNode node5 = new TreeNode(null, 5, null);        TreeNode node4 = new TreeNode(null, 4, null);        TreeNode node3 = new TreeNode(node6, 3, node7);        TreeNode node2 = new TreeNode(node4, 2, node5);        TreeNode node1 = new TreeNode(node2, 1, node3);        System.out.println("先序遍歷: ");        preOrder(node1);        System.out.println();        System.out.println("中序遍歷: ");        middleOrder(node1);        System.out.println();        System.out.println("後序遍歷: ");        lastOrder(node1);    }    /**     * 先序遍歷: 根結點->左孩子->右孩子  ==>擴展到整個二叉樹是: 根結點->左子樹->右子樹     * 1     * 2   3     * 先序就是: 1->2->3 看懂順序了吧     *     * @param node 根結點     */    public static void preOrder(TreeNode node) {        //最先訪問根結點        System.out.print(node.getData() + "  ");        if (!Objects.isNull(node.getLeftChildren())) {            preOrder(node.getLeftChildren());        } else {            if (!Objects.isNull(node.getRightChildren())) {                preOrder(node.getRightChildren());            } else {                return;            }        }        if (!Objects.isNull(node.getRightChildren())) {            preOrder(node.getRightChildren());        } else {            return;        }    }    /**     * 中序遍歷: 左孩子->根結點->右孩子  ==>擴展到整個二叉樹是: 左子樹->根結點->右子樹     * 1     * 2   3     * 中序就是: 2->1->3 看懂順序了吧     *     * @param node 根結點     */    public static void middleOrder(TreeNode node) {        if (!Objects.isNull(node.getLeftChildren())) {            middleOrder(node.getLeftChildren());        } else {            if (!Objects.isNull(node.getRightChildren())) {                middleOrder(node.getRightChildren());            } else {                System.out.print(node.getData() + "  ");                return;            }        }        //左子樹遍歷完之後訪問根結點        System.out.print(node.getData() + "  ");        if (!Objects.isNull(node.getRightChildren())) {            middleOrder(node.getRightChildren());        } else {            System.out.print(node.getData() + "  ");            return;        }    }    /**     * 後序遍歷: 左孩子->右孩子->根結點 ==>擴展到整個二叉樹是: 左子樹->右子樹->根結點     * 1     * 2   3     * 中序就是: 2->3->1 看懂順序了吧     *     * @param node 根結點     */    public static void lastOrder(TreeNode node) {        if (!Objects.isNull(node.getLeftChildren())) {            lastOrder(node.getLeftChildren());        } else {            System.out.print(node.getData() + "  ");            if (!Objects.isNull(node.getRightChildren())) {                lastOrder(node.getRightChildren());            } else {                return;            }        }        if (!Objects.isNull(node.getRightChildren())) {            lastOrder(node.getRightChildren());        } else {            return;        }        //左子樹->右子樹訪問完後->根結點        System.out.print(node.getData() + "  ");    }}
