Mysql修改字段名、字段類型、添加刪除字段、修改表名、表註釋

2020-10-29
| 科技
  • 修改字段名、字段類型、類型長度、默認值、註釋
    • ALTER  TABLE 表名 CHANGE 舊字段名 新字段名 新數據類型;   alter  table table1 change column1 column2 decimal(10,1) DEFAULT NULL COMMENT '註釋';-- 正確，能修改字段名、字段類型、類型長度、默認值、註釋alter  table table1 change column2 column2 varchar(100) DEFAULT 1.2 COMMENT '註釋'; -- 正確，此時字段名稱沒有改變，能修改字段類型、類型長度、默認值、註釋 alter  table table1 change column1 column2; -- 報錯

    使用change關鍵字，需要寫2個字段


    • 修改字段類型、類型長度、默認值、註釋
    ALTER  TABLE 表名 MODIFY [COLUMN] 字段名 新數據類型 新類型長度 新默認值 新註釋; -- COLUMN可以省略alter  table table1 modify column column1  decimal(10,1) DEFAULT NULL COMMENT '註釋'; -- 正確，能修改字段類型、類型長度、默認值、註釋alter  table table1 modify column1  decimal(10,2) DEFAULT NULL COMMENT '註釋'; -- 正確，能修改字段類型、類型長度、默認值、註釋

    使用modify關鍵字，寫1個字段即可


    • 修改表名
    ALTER TABLE 舊錶名 RENAME TO 新表名 ;mysql> show tables ;+-------------------+| Tables_in_db_test |+-------------------+| white_user   |+-------------------+1 row in set (0.00 sec) mysql> alter table white_user rename to white_user_new ;Query OK, 0 rows affected (0.00 sec) mysql> show tables ;+-------------------+| Tables_in_db_test |+-------------------+| white_user_new    |+-------------------+1 row in set (0.00 sec)


    • 修改表的註釋
    ALTER TABLE 表名 COMMENT '新註釋' mysql> alter table  white_user_new comment '新表-白名單表' ;Query OK, 0 rows affected (0.01 sec)Records: 0  Duplicates: 0  Warnings: 0 mysql> show create table white_user_new ; CREATE TABLE `white_user_new` (  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT 'ID',  `name` varchar(50) NOT NULL COMMENT '姓名',  `created_time` datetime DEFAULT NULL COMMENT '創建時間',  `updated_time` datetime DEFAULT NULL COMMENT '更新時間',  PRIMARY KEY (`id`)) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8 COMMENT='新表-白名單表'


    • 在指定位置添加新字段（如果不使用after，默認是在最後加字段）
    ALTER TABLE 表名 ADD [COLUMN] 字段名 字段類型 是否可為空 COMMENT '註釋' AFTER 指定某字段 ;--COLUMN關鍵字可以省略不寫mysql> alter table white_user_new add column erp varchar(50) not null comment 'erp賬號' after name ;Query OK, 0 rows affected (0.03 sec)Records: 0  Duplicates: 0  Warnings: 0--在name字段後面添加erp字段mysql> alter table white_user_new add position varchar(50) not null comment '崗位' after name ;Query OK, 0 rows affected (0.02 sec)Records: 0  Duplicates: 0  Warnings: 0--在name字段後面添加position字段。 mysql> show create table white_user_new ;                                      CREATE TABLE `white_user_new` (  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT 'ID',  `name` varchar(50) NOT NULL COMMENT '姓名',  `position` varchar(50) NOT NULL COMMENT '崗位',  `erp` varchar(50) NOT NULL COMMENT 'erp賬號',  `created_time` datetime DEFAULT NULL COMMENT '創建時間',  `updated_time` datetime DEFAULT NULL COMMENT '更新時間',  PRIMARY KEY (`id`)) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8 COMMENT='新表-白名單表'       mysql> alter table white_user_new add  mobile varchar(50) not null comment '手機號碼' before position ;--報錯，在position字段前添加mobile字段，不能使用before關鍵字


    • 刪除字段
    ALTER TABLE 表名 DROP [COLUMN] 字段名 ;-- COLUMN關鍵字可以省略不寫 mysql> alter table white_user_new drop column position ;Query OK, 0 rows affected (0.02 sec)Records: 0  Duplicates: 0  Warnings: 0-- 刪除position字段 mysql> alter table white_user_new drop erp ;Query OK, 0 rows affected (0.02 sec)Records: 0  Duplicates: 0  Warnings: 0-- 刪除erp字段mysql> show create table white_user_new ;                                      CREATE TABLE `white_user_new` (  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT 'ID',  `name` varchar(50) NOT NULL COMMENT '姓名',  `created_time` datetime DEFAULT NULL COMMENT '創建時間',  `updated_time` datetime DEFAULT NULL COMMENT '更新時間',  PRIMARY KEY (`id`)) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8 COMMENT='新表-白名單表'
