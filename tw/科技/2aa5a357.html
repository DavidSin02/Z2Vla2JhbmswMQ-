第二步：生成表單類

/*

* 表單類：

*

* 功能：根據不同的條件，輸出不同樣式的表單

*

* 核心思想：__toString()方法的使用

*

*/

class Form {

private $action;

private $shape;

function __construct($action='index.php'){

$this->action=$action;

$this->shape=isset($_REQUEST['action'])?$_REQUEST['action']:'rect';

}

function __toString(){

$form='<form action="'.$this->action.'" method="post">';

switch ($this->shape) {

case 'rect':

$form.=$this->getRect();

break;

case 'triangle':

$form.=$this->getTriangle();

break;

case 'circle':

$form.=$this->getCircle();

break;

default:

$form.='請選擇一個圖形';

break;

}

$form.='<input type="submit" name="sub" value="計算">';

$form.='</form>';

return $form;

}

private function getRect(){

$input='<b>請輸入矩形的寬度和高度:</b><p>';

$input.='<input type="hidden" name="action" value="'.$this->shape.'">';

$input.='長度：<input type="text" name="long" value="'.$_POST['long'].'"><p>';

$input.='寬度：<input type="text" name="width" value="'.$_POST['width'].'"><p>';

return $input;

}

private function getTriangle(){

$input='<b>請輸入三角形的三條邊:</b><p>';

$input.='<input type="hidden" name="action" value="'.$this->shape.'">';

$input.='第一條邊：<input type="text" name="side1" value="'.$_POST['side1'].'"><p>';

$input.='第二條邊：<input type="text" name="side2" value="'.$_POST['side2'].'"><p>';

$input.='第三條邊：<input type="text" name="side3" value="'.$_POST['side3'].'"><p>';

return $input;

}

private function getCircle(){

$input='<b>請輸入圓形的半徑:</b><p>';

$input.='<input type="hidden" name="action" value="'.$this->shape.'">';

$input.='半徑：<input type="text" name="radius" value="'.$_POST['radius'].'"><p>';

return $input;

}

}

第三步：生成結果類

class Result {

private $shape;

function __construct(){

switch ($_POST['action']) {

case 'rect':

$this->shape=new Rect();

break;

case 'triangle':

$this->shape=new Triangle();

break;

case 'circle':

$this->shape=new Circle();

break;

default:

$this->shape=false;

break;

}

}

function __toString(){

if($this->shape){

$result=$this->shape->shapeName.'的周長為：'.$this->shape->C().';面積為：'.$this->shape->S();

return $result;

}else {

return '沒有這個形狀';

}

}

}

第四步：生成抽象類——定義功能模塊開發規範

abstract class Shape {

var $shapeName;

abstract function S();

abstract function C();

protected function validate($value,$message){

if ($value=='' || !is_numeric($value) || $value < 0) {

echo '<font color="red">'.$message.'必須為非負數且不能為空！</font>';

return false;

}else {

return true;

}

}

}

第五步：定義各個圖像類

1. 矩形類

class Rect extends Shape{

private $width;

private $long;

function __construct(){

$this->shapeName='矩形';

if($this->validate($_POST['width'], '矩形的寬') && $this->validate($_POST['long'], '矩形的長')){

$this->width=$_POST['width'];

$this->long=$_POST['long'];

}else{

exit();

}

}

function C(){

$C=($this->long+$this->width)*2;

return $C;

}

function S(){

$S=$this->long*$this->width;

return $S;

}

}

class Triangle extends Shape{

private $a;

$this->shapeName='三角形';

if($this->check()){

$this->a=$_POST['side1'];

$this->b=$_POST['side2'];

}else {

exit();

}

}

function C(){

$C=$this->a+$this->b+$this->c;

return $C;

}

function S(){//利用"海倫公式"

$p=($this->a+$this->b+$this->c)/2;

$S=sqrt($p*($p-$this->a)*($p-$this->b)*($p-$this->c));

return $S;

}

3. 圓形類

class Circle extends Shape{

private $r;

function __construct(){

$this->shapeName='圓形';

if ($this->validate($_POST['radius'], '圓的半徑')) {

$this->r=$_POST['radius'];

}else {

exit();

}

}

function C(){

$C=2*round(pi(),0)*$this->r;

return $C;

}

function S(){

$S=round(pi(),0)*pow($this->r, 2);

return $S;

}

}
