# 文本處理，也就是樣本生成過程def text_processing(folder_path, test_size=0.2):    folder_list = os.listdir(folder_path)    # os.listdir 方法用於返回路徑下包含的文件或文件夾的名字的列表    # folder_list = ['C000008', 'C000014', 'C000013', 'C000022', 'C000023', 'C000024', 'C000010', 'C000020', 'C000016']        data_list = []    class_list = []    # 遍歷文件夾，每個文件夾裡是一個新聞的類別    for folder in folder_list:        new_folder_path = os.path.join(folder_path, folder)        # os.path.join就是把兩個路徑拼接        # new_folder_path = ./Database/SogouC/Sample/C000008                files = os.listdir(new_folder_path)        # 讀取new_folder_path路徑下的文件名        # ['15.txt', '14.txt', '16.txt', '17.txt', '13.txt', '12.txt', '10.txt', '11.txt', '19.txt', '18.txt']                # 讀取文件        j = 1        for file in files:            if j > 100:             # 怕內存爆掉，只取100個樣本文件，你可以註釋掉取完，            # 這裡每個類別下只有10個樣本，沒事                break            with open(os.path.join(new_folder_path, file), 'r') as fp:                raw = fp.read()                # read() 返回值為str，每次讀取整個文件，將文件所有內容放到一個字符串變量中                # readline() 返回值為str，每次只讀取一行,每行的內容放在一個字符串變量中                # readlines() 返回值為list，一次讀取整個文件，每行的內容放在一個字符串變量中作為列表的一個元素。                        ## 是的，隨處可見的jieba中文分詞            jieba.enable_parallel(4) # 開啟並行分詞模式，參數為並行進程數，不支持windows            word_cut = jieba.cut(raw, cut_all=False) # 精確模式，返回的結構是一個可迭代的genertor            word_list = list(word_cut) # genertor轉化為list，每個詞unicode格式            jieba.disable_parallel() # 關閉並行分詞模式                        data_list.append(word_list) #訓練集list            #class_list.append(folder.decode('utf-8')) #類別,str.decode會報錯            class_list.append(folder) #訓練集的標籤類別            j += 1                            ## 下面手動粗暴地劃分訓練集和測試集    data_class_list = zip(data_list, class_list) # zip 函數返回一個zip對象    data_class_list = list(data_class_list) # 需要用list轉換成列表        random.shuffle(data_class_list) # shuffle隨機打亂樣本    index = int(len(data_class_list)*test_size)+1    train_list = data_class_list[index:]    test_list = data_class_list[:index]        train_data_list, train_class_list = zip(*train_list)     # 解壓縮,文本和類別分開返回的是元組格式，可以在用list轉換    test_data_list, test_class_list = zip(*test_list)     #以上劃分訓練集和測試集其實可以用sklearn自帶的部分做    #train_data_list, test_data_list, train_class_list, test_class_list = \    #sklearn.model_selection.train_test_split(data_list, class_list, test_size=test_size)                # 統計詞頻放入all_words_dict    all_words_dict = {}    for word_list in train_data_list:        for word in word_list:            #if all_words_dict.has_key(word):已刪除此方法            if word in all_words_dict:                all_words_dict[word] += 1            else:                all_words_dict[word] = 1    # all_words_dict={'\n': 1257, '\u3000': 1986, '有': 175, '江湖': 1,.....}    # 同樣以上有現成的統計詞頻的API可以調用    # from collections import Counter     # Counter(train_data_list)    all_words_tuple_list = sorted(all_words_dict.items(), key=lambda f:f[1], reverse=True)     # 內建函數sorted第一個參數需為list，all_words_dict.items()轉化為列表，鍵和值為元組    # key函數代表按元組的的詞頻排序，並降序返回結果    # all_words_tuple_list = [('，', 3424), ('的', 2527), ('\u3000', 1734), ('。', 1482),.....]        #all_words_list = list(zip(*all_words_tuple_list)[0]) 報錯，需要修改    all_words_list,_ = zip(*all_words_tuple_list) # 解壓縮    all_words_list = list(all_words_list)    # all_words_list = ['，', '的', '\u3000', '。', '\n', '在', ' ', '、', '了', '“',.....]    return all_words_list, train_data_list, test_data_list, train_class_list, test_class_list

# 粗暴的詞去重def make_word_set(words_file):    words_set = set() # 集合格式    with open(words_file, 'r') as fp:        for line in fp.readlines(): # 循環取出每一行            word = line.strip()            # line.strip() 當()為空時，默認刪除空白符（包括'\n','\r','\t',' ')            if len(word)>0 and word not in words_set: # 去重                words_set.add(word)    return words_set

詞袋中選取有代表性的特徵詞

第一步生成的詞袋裡有很多通用的、無意義的詞語，需要去掉。
有代表性的詞語很大概率是一些對最終類別區分有作用的詞語。並且後面這些詞語會作為特徵作為模型的輸入。

def words_dict(all_words_list, deleteN, stopwords_set=set()):    # 選取特徵詞    feature_words = []    n = 1    for t in range(deleteN, len(all_words_list), 1):        # 循環時從第20個開始，也就是捨棄前20個詞語        if n > 1000: # feature_words的維度1000            break                    if not all_words_list[t].isdigit() and \        all_words_list[t] not in stopwords_set and \        1<len(all_words_list[t])<5:        # isdigit() 方法檢測字符串是否只由數字組成,返回True和False        # 滿足三個條件：不是數字；不在停用詞表；長度2～4            feature_words.append(all_words_list[t])            n += 1    return feature_words

# 文本特徵def text_features(train_data_list, test_data_list, feature_words, flag='nltk'):    def text_features(text, feature_words): # text的定義在下面        text_words = set(text) # 樣本去重        ## -----------------------------------------------------------------------------------        if flag == 'nltk':            ## nltk特徵 dict            features = {word:1 if word in text_words else 0 for word in feature_words}# 遍歷每個樣本詞語，凡是樣本的詞語出現在1000個特徵詞裡，就記錄下來，保存為字典格式，鍵為詞語，值為1，否則值為0。                    elif flag == 'sklearn':            ## sklearn特徵 list            features = [1 if word in text_words else 0 for word in feature_words] # 同上，遍歷每個樣本詞語，結果不是字典，出現即為1，不出現為0        else:            features = []        ## -----------------------------------------------------------------------------------        return features    train_feature_list = [text_features(text, feature_words) for text in train_data_list]    # text為每一個訓練的樣本，返回值是二維列表        test_feature_list = [text_features(text, feature_words) for text in test_data_list]    # train為每一個測試樣本，返回值是二維列表        return train_feature_list, test_feature_list

# 分類，同時輸出準確率等def text_classifier(train_feature_list, test_feature_list,                     train_class_list, test_class_list, flag='nltk'):    ## -----------------------------------------------------------------------------------    if flag == 'nltk':        ## 使用nltk分類器        train_flist = zip(train_feature_list, train_class_list)        train_flist = list(train_flist)         test_flist = zip(test_feature_list, test_class_list)        train_flist = list(test_flist)         classifier = nltk.classify.NaiveBayesClassifier.train(train_flist)        test_accuracy = nltk.classify.accuracy(classifier, test_flist)            elif flag == 'sklearn':        ## sklearn分類器        classifier = MultinomialNB().fit(train_feature_list, train_class_list)        # MultinomialNB()的使用方法和參數見：https://www.cnblogs.com/pinard/p/6074222.html                test_accuracy = classifier.score(test_feature_list, test_class_list)    else:        test_accuracy = []    return test_accuracy
