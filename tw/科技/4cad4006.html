二、C調用匯編的方法

1. 彙編顯示字符、清屏等子程序

[BITS 16]global io_hltglobal clear_screenglobal read_char;global print_char;global print_str;global shutdown;; 清屏clear_screen:    MOV ax, 03h    INT 10h	RET; 系統掛起io_hlt:	HLT	RET; 讀一個字符到在AL中read_char:    MOV ah,0x00    INT 0x16    RET; 打印AL中的字符print_char:    MOV AL,[esp+4]print_char_al:    MOV AH,0x0e    INT 0x10    CMP al, 0x0d    ; 是不是回車    JNE not_cr      ; 不是    MOV al, 0x0a    ; 輸出換行    INT 0x10        ; output the LF not_cr:     RET; 打印字符串的子程序，這裡沒有使用13H的BIOS子功能，而是使用循環顯示單個字符的方法，以後再優化print_str:    MOV edx,[esp+8]         ;len    MOV SI,[esp+4]          ;msgprint:    MOV      AL, [SI]    ADD      SI, 1    CMP      AL, 0          ; 字符串有沒有讀完，到0結束    DEC      EDX    JE       end_print      ; 跳到結束程序    MOV      AH, 0x0e       ; 顯示一個字符    MOV      BX, 15         ; 指定顏色    INT      0x10           ; 調用BIOS功能顯示字符    CMP      AL, 0x0d    JNE      not_cr_0    MOV      AL, 0x0a       ; 輸出換行    INT      0x10    MOV      BH, 0x00       ; 下面要把光標移到頭部，但沒起作用，以後再看是怎麼回事    MOV      AH, 0x02    MOV      DH, 0x01    MOV      DL, 0x01    INT      0x10not_cr_0:    JMP      printend_print:   RET; 關機，尚未調用shutdown:    MOV AX, 5301H        ;Function 5301h: APM Connect real-mode interface    XOR BX, BX               ;Device ID: 0000h (=system BIOS)    INT 15H                      ;Call interrupt: 15h    MOV AX, 530EH         ;Function 530Eh: APM Driver version    MOV CX, 0102H         ;Driver version: APM v1.2    INT 15H                       ;Call interrupt: 15h    MOV AX, 5307H        ;Function 5307h: APM Set system power state    MOV BL, 01H            ;Device ID: 0001h (=All devices)    MOV CX, 0003H        ;Power State: 0003h (=Off)    INT 15H                       ;Call interrupt: 15h    RET

2. 定義頭文件供C語言調用：

#ifndef __UTILITIES_H_#define __UTILITIES_H_// 此處只提供函數聲明void io_hlt(void);void clear_screen(void);char read_char(void);void print_char(char);void print_str(char *msg,int len);#endif

3. C獲取彙編返回值的方法

  • read_char的返回變量直接通過AX傳遞給C

4. 彙編獲取C參數的方法

  • 傳入C的變量，使用下面的方法獲取到C的參數：
print_str:    MOV edx,[esp+8]         ;len    MOV SI,[esp+4]          ;msg
