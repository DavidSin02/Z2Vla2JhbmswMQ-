mybatis模糊查找

2020-11-14
| 科技

方式1：xml配置

接口調用userMapper.findByName("c")

<select id="findByName" parameterType="String" resultMap="TUserMap">

SELECT * FROM t_user WHERE nick_name like concat('%',#{nickName},'%')

</select>

方式2：xml配置

接口調用userMapper.findByName("c")

<select id="findByName" parameterType="String" resultMap="TUserMap">

<bind name="nickName" value="'%'+nickName+'%'"/>

SELECT * FROM t_user WHERE nick_name like #{nickName}

</select>

方式3：xml配置

接口調用userMapper.findByName("%c%")

<select id="findByName" parameterType="String" resultMap="TUserMap">

SELECT * FROM t_user WHERE nick_name like #{nickName}

</select>

方式4：xml配置

接口調用userMapper.findByName("c")

<select id="findByName" parameterType="String" resultMap="TUserMap">

SELECT * FROM t_user WHERE nick_name like '%${nickName}%'

</select>

方式4：註解

接口調用userMapper.findByName("c")

@Select("select * from t_user where nick_name like '%${nickName}%'")

List<TUser> findByNameAnnotation(String nickName);

方式5：註解

接口調用userMapper.findByName("%c%")

@Select("select * from t_user where nick_name like #{nickName}")

List<TUser> findByNameAnnotation(String nickName);
