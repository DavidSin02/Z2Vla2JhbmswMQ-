from pyspider.libs.base_handler import *

class Handler(BaseHandler):

crawl_config = {

}

@every(minutes=24 * 60)

def on_start(self):

self.crawl(

'http://www.cdwater.gov.cn/admin/newsData.proxy?jsoncallback=jQuery18306603365339985892_1530710618747&channelId=0101001002000&pageNo=1&maxNo=12&citycode=&length=36&tag=1&username=%25E4%25B8%25AD%25E6%2596%2587',

callback=self.index_page, validate_cert=False)#初始的第一個動態url

@config(age=10 * 24 * 60 * 60)

def index_page(self, response):

print(response)

web_data = response.text#返回的數據

data = re.findall(r'[^()]+', web_data)[1]#數據處理，得到需要的json數據串

content_list = json.loads(data)#得到json數據

url = []

for i in range(0, len(content_list['content_list'])):#從json數據中獲取url，title

url.append(content_list['content_list'][i]['url'])

print(content_list['content_list'][i]['url'])

self.crawl(content_list['content_list'][i]['url'], callback=self.detail_page, validate_cert=False)

current_page = 1 # 當前頁數

total_page = 10 # 總頁數

next_page = current_page + 1 # 下一頁

if next_page <= total_page: # 獲取下一頁的數據構造翻頁

self.crawl(

'http://www.cdwater.gov.cn/admin/newsData.proxy?jsoncallback=jQuery18306603365339985892_1530710618747&channelId=0101001002000&pageNo={}&maxNo=12&citycode=&length=36&tag=1&username=%25E4%25B8%25AD%25E6%2596%2587'.format(

str(next_page)), callback=self.index_page)

@config(priority=2)

def detail_page(self, response):

return {

"url": response.url,

"title": response.doc('title').text(),

}

我們導入了2個模塊他們的作用分別是：re用於正則獲取數據，json用於json數據轉化

開始異步爬蟲分析：

第一步：瀏覽器打開需要爬取的url：

按F12使用谷歌瀏覽器自帶的瀏覽器network工具，查看請求，或者自己使用抓包工具也可以（當前沒有請求，莫方，你手動去點擊下一頁製造請求）：

點擊下一頁或者直接刷新，就會在network上邊看到有請求，這裡注意看XHR和JS部分，有的是用js封數據請求會在js裡面，也有會在XHR裡面的異步請求：

從這裡我們就可以清楚的在xhr返回的header裡面看到異步請求的url，這裡我們直接訪問該url（或者在preview裡面可以看到返回的數據）可以看到返回的數據就是新聞數據，也就是我們要爬的數據。

這下我們知道數據是通過這樣式的url傳過來的，因此我們需要分析url格式，不要怕一般都很有規律性，都是根據下邊的參數構造的，找到規律就能構造出url出來，這裡不詳細分析，具體看目標網站如何封裝。

到這裡，我們基本知道接下來就是根據構造的url去寫爬蟲腳本，做詳細的數據解析，這個就不詳細講了，具體問題具體分析。

這就是一種解決動態/異步請求html數據爬取的一種思路，有其他思路可以一起溝通。
