mysql快速複製表結構的幾種方法

2020-11-14
| 科技

有時候我們需要原封不動的複製一張表的表結構來生成一張新表，MYSQL可以過過以下幾種方法快速複製表結構。

假設已存在表t_test：

//複製表t_test生成t_test_b

create table t_test_b like t_test;

//複製表結構前先檢測表是否已存在

create table if not exists t_test_like (like t_test);

//獲取創建表的sql語名，然後修改表名(AUTO_INCREMENT等)後執行sql語名

show create table t_test

//快速複製表結構

create table t_test_c select * from t_test WHERE 1=2

create table t_test_c select * from t_test limit 0

//快速複製表結構及數據

create table t_test_c select * from t_test
