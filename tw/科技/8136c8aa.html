C｜結構和結構指針做為函數參數或返回值

2020-11-14
| 科技

函數的一般形式：

函數類型說明型 函數名(形式參數列表){}

（當然也可以不使用參數）

根據有無返回值，可以分為：

void 函數名(形式參數列表){......}

數據類型 函數名(形式參數列表)

{

......

return語句;

}

return語句返回的數據類型必須與函數名前面聲明的數據類型一致；

函數定義時定義形參，函數調用時使用實參，形參相當於變量聲明，實參相當於變量賦值，實參可以是常量、變量、表達式、函數，但必須具有確定的值。形參和實參的類型應相同或賦值兼容。

形參變量只有在被調用時才分配內存單元，在調用結束後會釋放所分配的內存單元。因此，形參只有在函數內部有效。函數調用結束後不能再使用該形參變量。

函數參數和返回類型可以是標量變量（聲明為基本類型的變量），也可以是複合變量（結構體等構造類型的變量）。可以是變量本身，也可以是變量指針（指針傳遞的結構變量地址值是其首地址。）

//返回結構及結構指針的例子。

#include <stdio.h>

#include <stdlib.h>

struct LIST{

int a,b;

}data[3],*p;

//返回結構值

struct LIST make1(int x, int y)

{

struct LIST temp;

temp.a=x;

temp.b=y;

return temp;

}

//返回結構指針

struct LIST *make2(int x, int y)

{

struct LIST *temp;

temp=(struct LIST *)malloc(sizeof(struct LIST));

temp->a=x;

temp->b=y;

return temp;

}

int main()

{

int i;

data[0]=make1(325,8431);

p=make2(855,56);

data[1]=*p;

data[2]=data[0]; //結構可以整體賦值

for(i=0;i<3;i++)

{

printf("data[%d].a=%d,data[%d].b=%d\n", i,data[i].a,i,data[i].b);

}

system("pause");

return 0;

}

運行結果：

data[0].a=325,data[0].b=8431

data[1].a=855,data[1].b=56

data[2].a=325,data[2].b=8431

//使用結構參數和結構指針參數。

#include <stdio.h>

struct LIST{

int a,b;

}data[3];

struct LIST make(int , int); //返回結構的函數

void disp(struct LIST [ ]); //使用結構參數

void display(struct LIST *); //使用結構指針參數

int main()

{

data[0]=make(133,138);

data[1]=make(185,180);

data[2]=make(136,182);

disp(data); //這裡的data以結構參數出現

display(data); //這裡的data以結構的首地址出現，即結構指針參數

system("pause");

return 0;

}

struct LIST make(int x, int y)

{

struct LIST temp;

temp.a=x;

temp.b=y;

return temp;

}

void disp(struct LIST data[ ])

{

int i;

for(i=0;i<3;i++)

{

printf("data[%d].a=%d,data[%d].b=%d\n", i,data[i].a,i,data[i].b);

}

}

void display(struct LIST *p)

{

int i;

for(i=0;i<3;i++)

{

printf("data[%d].a=%d,data[%d].b=%d\n", i,(p+i)->a,i, (p+i)->b);

}

}

運行結果：

data[0].a=133,data[0].b=138

data[1].a=185,data[1].b=180

data[2].a=136,data[2].b=182

data[0].a=133,data[0].b=138

data[1].a=185,data[1].b=180

data[2].a=136,data[2].b=182

//結構體作為函數參數

#include <stdio.h>

struct time

{

long int day;

long int hour;

}

main ()

{

struct time minute;

printf ("input day:\n");

scanf ("%ld",&minute.day);

printf ("input hour:\n");

scanf ("%ld",&minute.hour);

print (minute);

system("pause");

}

print (struct time minute)

{

long int i;

i=(minute.day*24+minute.hour)*60;

printf ("total minute is:%ld\n",i);

}

運行結果

input day:

100

input hour:

10

total minute is:144600

//結構體指針作為函數參數

#include <stdio.h>

struct time

main ()

}

}
