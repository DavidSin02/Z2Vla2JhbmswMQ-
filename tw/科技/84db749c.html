JavaScript高級程序設計第六章面向對象程序設計

2020-10-29
| 科技

數據屬性

[[configurable]]：表示能否通過delete刪除屬性從而重新定義屬性

[[Enumerable]]：表示能夠通過for-in循環數據

[[Writable]]：表示能夠通過修改屬性的值

[[Value]]：包含這個屬性的數據值

  1. //"use strict"//嚴格模式應該寫在函數開頭位置

  2. var person = {

  3. name:'李森',

  4. age:27

  5. };

  6. Object.defineProperty(person,"name",{

  7. writable:false,//定義變量不可修改

  8. value:'lisen'//設置變量值

  9. });

  10. person.name='lisen1';//啟用嚴格模式會報錯，非嚴格模式會忽略修改

  11. console.log(person.name);

  12. Object.defineProperty(person,'name',{

  13. configurable:false

  14. });

  15. delete person.name;//刪除屬性

  16. console.log(person.name);//啟用嚴格模式會報錯，非嚴格模式會忽略更改

訪問屬性

常見的包括get和set屬性

[[get]]：在獲取屬性時調用的函數

[[set]]：在設置屬性時調用的函數

  1. var animal ={

  2. _name:"Tiger",

  3. _amount:10

  4. };

  5. Object.defineProperty(animal,"name",{

  6. get:function () { returnthis._name;},

  7. set:function (newValue) {

  8. this._amount+=1;

  9. this._name=newValue;

  10. }

  11. });

  12. console.log(animal.name);

  13. animal.name="Sheep";

  14. console.log("開始輸出_amount");

  15. console.log(animal._amount);

  16. console.log("開始輸出_name");

  17. console.log(animal._name);

批量設置屬性值

  1. var animal ={

  2. _name:"Tiger",

  3. _amount:10

  4. };

  5. Object.defineProperties(animal,{

  6. name:{

  7. get:function () {

  8. returnthis._name;

  9. },

  10. set:function (newValue) {

  11. this._name=newValue;

  12. }

  13. },

  14. amount:{

  15. value:200

  16. }

  17. });

  18. console.log(animal.amount);//200

獲取屬性值

  1. var desc = Object.getOwnPropertyDescriptor(animal,"name");

  2. console.log(desc.value);

構造函數

其實構造函數與其他函數沒有什麼本質的區別，任何函數通過new關鍵字都可以作為構造函數。

  1. function Person(name) {

  2. this.name=name;

  3. this.sayName=function () {

  4. alert(name);

  5. }

  6. }

  7. var person = new Person("123");

  8. person.sayName();

  9. console.log(person.name);

通用做法

一般情況下，我們可以通過匿名函數進行創建

  1. var obj = obj || {};

  2. (function (o,$) {

  3. o.name = "lisen";

  4. $.obj = obj;

  5. })(obj,jQuery);

  6. alert(obj.name);

  7. alert($.obj.name);
