如何用LCD1602的字符來顯示自己名字的拼音

2020-11-14
| 科技

SMC1602什麼意思？長什麼樣子

各個接口管腳的定義；

基本操作時序：

讀狀態 輸入：RS=L,R/W=H,E=H 輸出D0~D7 =狀態字

讀數據 輸入：RS=H,R/W=H,E=H 輸出D0~D7 =數據

寫指令 輸入：RS=L,R/W=L,E=H,D0~D7=指令碼 輸出：無

寫數據 輸入：RS=H,R/W=LE=H,D0~D7=數據 輸出：無

RAM地址映射圖

向圖中00H~0FH,40H~4F地址任意寫顯示數據時，液晶都可以立即顯示出來，但是10H-27H或者50H-67H地址時，就要用到屏幕移動指令，把它們移動到顯示區域：

控制器內部有個數據地址指針，用戶可以通過聽你們訪問內部的全部字節。

使用方法：0X80+0X要寫入的地址。

清屏指令：0x01;

為什麼是這樣的指令，這是廠家自定義 定義好的，不能改變。

還有一個是顯示模式設置：0x38; 意思是：設置16*2顯示，5*7點陣，8位數據接口。（屬於初始化步驟裡面的）

顯示開關以及光標設置

詳細解釋一下：

#include "reg52.h" //包含51系列單片機的一個頭文件

#define uchar unsigned char//宏定義，把後面的長句，變成前面的短句

#define uint unsigned int //宏定義，把後面的長句，變成前面的短句

uchar code table[]="Huang Hua Nan"; //編碼來顯示的數據，huang hua nan 13

uchar code table1[]="2019/06/02"; //編碼 顯示的字符，是 今天的日期。 10

sbit lcden=P3^4; //位操作，把3.4這個口定義為lcden,

sbit lcdrs=P3^5;//位操作，把3.5這個口定義為lcdrs,

sbit lcdwr=P3^6;//位操作，把3.6這個口定義為lcdwr,

uchar num; //定義一個無符號字符。

void delayms(uint xms) //延時子程序，定義了一個無符號字符，帶形式參數

{

uint i,j; //定義兩個無符號字符

for(i=xms;i>0;i--)//要循環xms次

for(j=110;j>0;j--); //要循環xms*110

}

void write_com(uchar com) //寫指令子程序 意義 ：先啟動寫命令操作，然後把具體寫到哪裡去，再調用。

{

lcdrs=0; //寫指令，寫命令。 低電平有效

lcdwr=0; // 低電平有效。

P1=com; //把寫的指令，送到P1口

delayms(5); //延時5毫秒

lcden=1;//寫命令要高電平

delayms(5);//延時5毫秒

lcden=0;//變成低電平，表示寫完了。

}

void write_date(uchar date) //寫數據子程序， 送要顯示的數據。

{

lcdrs=1; //高電平有效

lcdwr=0; //低電平。

P1=date; //把要送去顯示的數據 送到p1口

delayms(5);//延時5毫秒

lcden=1; //高電平有效，

delayms(5);//延時5毫秒

lcden=0;//告訴系統，顯示的數據 我已經送完了。

}

void init() //初始化子程序

{

lcden=0; //根據操作時序來看，使能端要低電平。

write_com(0x38);//顯示模式設置，也就是00111000 ，系統自定義的，有興趣的可以去查閱更細節的資料。

write_com(0x0e); //光標打開，顯示光標，不閃爍。 00001110

write_com(0x06); //寫入一個字符後指針地址加一，寫一個字符時整屏不移動 00000110

write_com(0x01); //清屏顯示，數據指針清零，所以顯示清零 0000 0001

write_com(0x80); //設置字符顯示首地址。 1000 0000 +01 確認從屏幕的第幾位開始顯示

}

void main()//主程序

{

init(); //初始化

delayms(100);//延時100毫秒，給初始化一點時間。

EA=1;//開總中斷

EX0=1;//外部中斷0， 來顯示數據

EX1=1;//外部中斷1， 清屏

while(1);//不斷地判斷，循環。

}

void int0() interrupt 0 外部中斷0

{

write_com(0x80+0x01); 寫命令，指令，在第二位開始顯示

for(num=0;num<14;num++)// huang hua nan 因為有13個數據，所以要寫14.

{

write_date(table[num]); //調用顯示數據子程序

delayms(20);//延時20毫秒

}

write_com(2);//顯示回車 換行的意思。

write_com(0x80+0x41); 去第二行第二個位置去顯示。

for(num=0;num<11;num++) //因位日期數據是10位，所以我們寫11.

{

write_date(table1[num]);//調用顯示數據子程序，送去顯示

delayms(20);//延時20毫秒

}

}

void int1() interrupt 2 第二個中斷

{

write_com(1); 表示清屏的意思。

}
