OkHttp3

基於OkHttp3封裝的網絡請求工具類

DEMO下載地址：https://github.com/MrZhousf/OkHttp3

功能點

  • 支持Http/Https等協議
  • 支持Cookie持久化
  • 支持Gzip壓縮
  • 支持協議頭參數Head設置、二進制參數請求
  • 支持Unicode自動轉碼、服務器響應編碼設置
  • 支持同步/異步請求
  • 支持四種緩存類型請求：僅網絡、僅緩存、先網絡再緩存、先緩存再網絡
  • 支持自定義緩存存活時間與緩存清理功能
  • 當Activity/Fragment銷燬時自動取消相應的所有網絡請求，支持取消指定請求
  • 異步請求響應自動切換到UI線程，摒棄runOnUiThread
  • Application中自定義全局配置/增加系統默認配置
  • 支持文件和圖片上傳/批量上傳，支持同步/異步上傳，支持進度提示
  • 支持文件下載/批量下載，支持同步/異步下載，支持進度提示
  • 支持文件斷點下載，獨立下載的模塊摒棄了數據庫記錄斷點的過時方法
  • 完整的日誌跟蹤與異常處理
  • 支持請求結果攔截以及異常處理攔截
  • 支持單例客戶端，提高網絡請求速率
  • 後續優化中...

自定義全局配置

在Application中配置如下：

String downloadFileDir = Environment.getExternalStorageDirectory().getPath()+"/okHttp_download/";

String cacheDir = Environment.getExternalStorageDirectory().getPath();

if(getExternalCacheDir() != null){

//緩存目錄，APP卸載後會自動刪除緩存數據

cacheDir = getExternalCacheDir().getPath();

}

OkHttpUtil.init(this)

.setConnectTimeout(15)//連接超時時間

.setWriteTimeout(15)//寫超時時間

.setReadTimeout(15)//讀超時時間

.setMaxCacheSize(10 * 1024 * 1024)//緩存空間大小

.setCacheType(CacheType.FORCE_NETWORK)//緩存類型

.setHttpLogTAG("HttpLog")//設置請求日誌標識

.setIsGzip(false)//Gzip壓縮，需要服務端支持

.setShowHttpLog(true)//顯示請求日誌

.setShowLifecycleLog(false)//顯示Activity銷燬日誌

.setRetryOnConnectionFailure(false)//失敗後不自動重連

.setCachedDir(new File(cacheDir,"okHttp_cache"))//緩存目錄

.setDownloadFileDir(downloadFileDir)//文件下載保存目錄

.setResponseEncoding(Encoding.UTF_8)//設置全局的服務器響應編碼

.addResultInterceptor(HttpInterceptor.ResultInterceptor)//請求結果攔截器

.addExceptionInterceptor(HttpInterceptor.ExceptionInterceptor)//請求鏈路異常攔截器

.setCookieJar(new PersistentCookieJar(new SetCookieCache(), new SharedPrefsCookiePersistor(this)))//持久化cookie

.build();

取消指定請求

建議在視圖中採用OkHttpUtil.getDefault(this)的方式進行請求綁定，該方式會在Activity/Fragment銷燬時自動取消當前視圖下的所有請求； 請求標識類型支持Object、String、Integer、Float、Double； 請求標識務必保證唯一。

//*******請求時先綁定請求標識，根據該標識進行取消*******/

//方法一：

OkHttpUtil.Builder()

.setReadTimeout(120)

.build("請求標識")//綁定請求標識

.doDownloadFileAsync(info);

//方法二：

OkHttpUtil.getDefault("請求標識")//綁定請求標識

.doGetSync(info);

//*******取消指定請求*******/

OkHttpUtil.getDefault().cancelRequest("請求標識");

/**

* 同步請求：由於不能在UI線程中進行網絡請求操作，所以採用子線程方式

*/

private void sync() {

new Thread(new Runnable() {

@Override

public void run() {

final HttpInfo info = HttpInfo.Builder()

.setUrl(url)

.setResponseEncoding(Encoding.UTF_8)//設置服務器響應編碼

.build();

OkHttpUtil.getDefault(this)

.doGetSync(info);

final String result = info.getRetDetail();

runOnUiThread(new Runnable() {

@Override

public void run() {

resultTV.setText("同步請求：" + result);

setFromCacheTV(info);

}

});

}

}).start();

}

private void tenSecondCache(){

請求結果統一預處理攔截器/請求鏈路異常信息攔截器示例

請求結果攔截器與鏈路異常攔截器方便項目進行網絡請求業務時對信息返回的統一管理與設置

/**

* Http攔截器

* 1、請求結果統一預處理攔截器

* 2、請求鏈路異常信息攔截器

* @author zhousf

*/

public class HttpInterceptor {

/**

* 請求結果統一預處理攔截器

* 該攔截器會對所有網絡請求返回結果進行預處理並修改

*/

public static ResultInterceptor ResultInterceptor = new ResultInterceptor() {

@Override

public HttpInfo intercept(HttpInfo info) throws Exception {

//請求結果預處理：可以進行GSon過濾與解析

return info;

}

};

/**

* 請求鏈路異常信息攔截器

* 該攔截器會發送網絡請求時鏈路異常信息進行攔截處理

*/

public static ExceptionInterceptor ExceptionInterceptor = new ExceptionInterceptor() {

@Override

public HttpInfo intercept(HttpInfo info) throws Exception {

switch (info.getRetCode()){

case HttpInfo.NonNetwork:

info.setRetDetail("網絡中斷");

break;

case HttpInfo.CheckURL:

info.setRetDetail("網絡地址錯誤["+info.getNetCode()+"]");

break;

case HttpInfo.ProtocolException:

info.setRetDetail("協議類型錯誤["+info.getNetCode()+"]");

break;

case HttpInfo.CheckNet:

info.setRetDetail("請檢查網絡連接是否正常["+info.getNetCode()+"]");

break;

case HttpInfo.ConnectionTimeOut:

info.setRetDetail("連接超時");

break;

case HttpInfo.WriteAndReadTimeOut:

info.setRetDetail("讀寫超時");

break;

case HttpInfo.ConnectionInterruption:

info.setRetDetail("連接中斷");

break;

}

return info;

}

};

}

case R.id.force_network_btn://僅網絡

forceNetwork();

break;

case R.id.force_cache_btn://僅緩存

forceCache();

break;

case R.id.network_then_cache_btn://先網絡再緩存

networkThenCache();

break;

case R.id.cache_then_network_btn://先緩存再網絡

cacheThenNetwork();

break;

case R.id.ten_second_cache_btn://緩存10秒失效

tenSecondCache();

break;

case R.id.delete_cache_btn://清理緩存

deleteCache();

break;

}

}

/**

* 同步請求：由於不能在UI線程中進行網絡請求操作，所以採用子線程方式

*/

private void sync() {

new Thread(new Runnable() {

@Override

public void run() {

final HttpInfo info = HttpInfo.Builder()

.setUrl(url)

.setResponseEncoding(Encoding.UTF_8)//設置服務器響應編碼

.build();

OkHttpUtil.getDefault(this)

.doGetSync(info);

final String result = info.getRetDetail();

runOnUiThread(new Runnable() {

@Override

public void run() {

resultTV.setText("同步請求：" + result);

setFromCacheTV(info);

}

});

}

}).start();

needDeleteCache(true);

}

/**

* 異步請求：回調方法可以直接操作UI

*/

private void async() {

OkHttpUtil.getDefault(this).doGetAsync(

HttpInfo.Builder().setUrl(url).build(),

new Callback() {

@Override

public void onFailure(HttpInfo info) throws IOException {

String result = info.getRetDetail();

resultTV.setText("異步請求失敗：" + result);

}

@Override

public void onSuccess(HttpInfo info) throws IOException {

String result = info.getRetDetail();

resultTV.setText("異步請求成功：" + result);

//GSon解析

TimeAndDate time = new Gson().fromJson(result, TimeAndDate.class);

LogUtil.d("MainActivity", time.getResult().toString());

setFromCacheTV(info);

}

});

needDeleteCache(true);

}

/**

* 緩存10秒失效：連續點擊進行測試10秒內再次請求為緩存響應，10秒後再請求則緩存失效並進行網絡請求

*/

private void tenSecondCache(){

//由於採用同一個url測試，需要先清理緩存

if(isNeedDeleteCache){

isNeedDeleteCache = false;

OkHttpUtil.getDefault().deleteCache();

}

OkHttpUtil.Builder()

.setCacheType(CacheType.CACHE_THEN_NETWORK)

.setCacheSurvivalTime(10)//緩存存活時間為10秒

.build(this)

.doGetAsync(

HttpInfo.Builder().setUrl(url).build(),

new Callback() {

@Override

public void onSuccess(HttpInfo info) throws IOException {

String result = info.getRetDetail();

resultTV.setText("緩存10秒失效：" + result);

setFromCacheTV(info);

}

@Override

public void onFailure(HttpInfo info) throws IOException {

resultTV.setText("緩存10秒失效：" + info.getRetDetail());

}

}

);

}

private void needDeleteCache(boolean delete){

isNeedDeleteCache = delete;

}

private void setFromCacheTV(HttpInfo info){

fromCacheTV.setText(info.isFromCache()?"緩存請求":"網絡請求");

}

* 清理緩存

private void deleteCache(){

if(OkHttpUtil.getDefault().deleteCache()){

ToastUtil.show(this,"清理緩存成功");

}else{

ToastUtil.show(this,"清理緩存失敗");

}
