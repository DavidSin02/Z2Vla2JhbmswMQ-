概述

一鍵部署rsync

核心部分如下：

# auto install rsync on el7function install_rsync3_el7() {  echo ""  echo -e "\033[33m****************************************************安裝Rsync 3.1.2*****************************************************\033[0m"   #下載包  if [ -f /opt/rsync*.tar.gz ] ;then	  echo "*****存在rsync安裝包，無需下載*****"  else      ping -c 4 download.samba.org >/dev/null 2>&1      if [ $? -eq 0 ];then      	wget https://download.samba.org/pub/rsync/src/rsync-3.1.2.tar.gz -O /opt/rsync-3.1.2.tar.gz      else        echo "please download Rsync-3.1.2 package manual !"		exit $?      fi   fi  #安裝rsync 3.1.2  action "********編譯安裝rsync 遠程同步服務********" /bin/true  chmod u+x /opt/rsync*.tar.gz && tar -xvf /opt/rsync*.tar.gz >/dev/null 2>&1 &    cd /opt/rsync-3.1.2  &>/dev/null && ./configure --prefix=${RSYNC_HOME}  &>/dev/null && make &>/dev/null  && make install &>/dev/null     #修改rsync配置文件  cp ${RSYNC_HOME}/rsyncd.conf ${RSYNC_HOME}/rsyncd.conf_bak  &>/dev/null && mkdir -p /opt/rsync &>/dev/null  cat > ${RSYNC_HOME}/rsyncd.conf << EOF #全局參數uid = rootgid = rootuse chroot = nomax connections = 5strict modes = yesport = 8888pid file = /var/run/rsyncd.pidlock file = /var/run/rsync.locklog file = /var/log/rsyncd.log#模塊[jpom]path = /opt/rsynccomment = jpom deployignore errorsread only = falselist = nohosts allow = 0.0.0.0auth users = rsyncsecrets file = ${RSYNC_HOME}/rsyncd.secretsEOFcat > ${RSYNC_HOME}/rsyncd.secrets << EOF${RSYNC_USER}:${RSYNC_PWD}EOF  #授權  chmod -R 755 ${RSYNC_HOME} && chmod 600 ${RSYNC_HOME}/rsyncd.secrets  #啟動rsync  action "********啟動rsync********" /bin/true  ${RSYNC_HOME}/bin/rsync --daemon --config=${RSYNC_HOME}/rsyncd.conf  echo -e "\033[33m**************************************************完成Rsync 3.1.2安裝***************************************************\033[0m"cat > /tmp/rsync.log << EOFrsync版本：   3.1.2rsync目錄：   ${RSYNC_HOME}rsync用戶名： ${RSYNC_USER}rsync密碼：   ${RSYNC_PWD}rsync密碼：   ${RSYNC_PORT}EOF  cat /tmp/rsync.log  echo -e "\e[1;31m 以上信息10秒後消失，保存在/tmp/rsync.log文件下 \e[0m"  echo -e "\033[33m************************************************************************************************************************\033[0m"  echo ""  sleep 10}
