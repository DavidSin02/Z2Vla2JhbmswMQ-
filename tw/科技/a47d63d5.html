MASM 彙編語言——顯示彩色字符

編程：在屏幕中間顯示80行若干種（可以隨時更換，位於color segment）顏色的字符串(可更換位於data segment)，主要學習數據段、指令段和棧的使用，編寫簡單的calll , ret模塊調用。

assume cs:code,ds:data,ss:stack;棧段stack segment dw 0,0,0,0,0,0,0,0,0stack ends;打印的數據段;一般屏幕寬度為160byte，每2byte顯示一個字符，其中低位是ASCII碼，高位是字符屬性;一共可以顯示80個字母或數字字符;字符數可以更改data segment  db 'Welcome to Masm!',0 ;以0結尾符  ;db 4ah,71h,4fh,02h,31h,3ah,0 data ends ;字符顏色屬性數據段;每一行字符一種顏色屬性; 7 6 5 4 3 2 1 0;閃爍 R G B 高亮 R G B; 背景 前景;R:紅色 G：綠色 B：藍色 按位設置屬性字節，配出各種不同color segment  db 4ah,71h,4fh,02h,31h,3ah,42h,7ah,7fh,41h,0 ;以0結尾符color ends;代碼段code segmentstart: mov ax,data ;data,color,stack段初始化 mov ds,ax mov ax,color mov es,ax mov ax,stack mov ss,ax mov bx,0 mov si,0 mov di,0a0h ;add di,2 mov ax,0 mov cx,0 lp: ;獲取字符長度 cmp byte ptr [si],0 je s inc si inc cx jmp lp  s:  mov bx,0 mov ax,cx ;mov si,ax ;mov dl,2 ;div dl mov dx,80 ;計算顯示字符開始空格數 sub dx,ax ;mov dh,0 add di,dx call s3 mov bx,0 mov cx,80 ;顯示80行數據  s1: cmp bx,dx ;字符顏色種類循環使用 jne s2 mov bx,0   s2: mov ah,byte ptr es:[bx]；獲取字符顏色 mov al,0 inc bx push bx push di push cx mov cx,si mov bx,0  s0:  mov dl,ds:[bx] ;顯示一行字符 mov dh,ah push ax mov ax,0B800h push ds mov ds,ax mov ds:[di],dx pop ds pop ax inc bx add di,2  loop s0  pop cx pop di add di,0a0h ;add di,0a0h ;pop bx call s3 pop bx  loop s1  ;程序結束，控制權回操作系統 mov ax,4c00h int 21h ;獲得字符顏色屬性種類  ;get color number ;register (ax) ;retur(dx)   s3: mov bx,0 mov dx,0 s4: cmp byte ptr es:[bx],0 je s5  inc dx inc bx jmp s4  s5: mov bx,0  ret   code ends end start
