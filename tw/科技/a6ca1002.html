Java中Collection集合ArrayList和HashSet的使用

2020-11-14
| 科技
package dao;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashSet;
/*
* 本類講解java中的Collection集合的使用
*/
public class CollectionDemo {
public static void main(String[] args) {
//要使用Collection集合 第一步需要創建Collection的對象
//ArrayList是Collection的一個實現類
//Collection有很多個實現了 ArrayList是其中的一種
//要使用以下的兩個類得先導入包
System.out.println("-------以下是用ArrayList子類來創建集合對象------");
Collection demo1 = new ArrayList();
//對象創建好了 接下來就是往對象裡面放東西了
//注意 ：集合中可以存放任意類型的數據，因為Collection會把這些統一識別為object對象
demo1.add("北京");//用提供的add方法像集合中添加內容
demo1.add("上海");
demo1.add(100);
demo1.add("天津");
//獲取集合中的個數
System.out.println("demo1的元素個數為："+demo1.size());//輸出集合中的個數
//輸出集合中的所有值
//這個地方可以看到輸出的值為一個連續的值，吧集合中所有的值都輸出了
//原因是 因為 Collection集合的所有子類都重寫了toString方法。該方法可以一次性的輸出集合中的所有元素
System.out.println("移除天津之前的值為："+demo1);
//移除某個指定的元素 對象名.remove(元素值)
demo1.remove("天津");
//輸出移除之後的
System.out.println("移除天津之後的值為："+demo1);
//判斷集合中是否存在某個指定的元素 contains
System.out.println("demo1集合中是否存在上海字符串："+demo1.contains("上海"));
System.out.println("demo1集合中是否存在廣州字符串："+demo1.contains("廣州"));
//一次性刪除demo1集合中的所有元素
demo1.clear();
//clear之後的值為空值 因為集合被clear之後就清空了所有的元素
System.out.println("clear demo1集合之後的值為："+demo1);
System.out.println("\n-------以下是用HashSet子類來創建集合對象------");
//創建第二個Collection對象 用子類HashSet實現
//同樣也需要導入HashSet的包
Collection demo2 = new HashSet();
demo2.add("海淀區");
demo2.add("朝陽區");
demo2.add("東城區");
demo2.add("西城區");
//輸出
System.out.println("移除海淀區之前的值為:"+demo2);
demo2.remove("海淀區");
System.out.println("移除海淀區之後的值為:"+demo2);
//判斷一個集合是否完全包含另一個集合 結果為false
System.out.println("demo1集合是否包含demo2集合:"+demo1.containsAll(demo2));
//小結：在該案例裡面主要講解了Collection接口用它的兩個實現類來創建對象
// 雖然demo1和demo2是兩個不同的實現類創建的對象
// 但是卻擁有相同的方法 ，因為它們都同時是Conllction的子類
// 所以在使用add size remove clear contains時沒有任何的區別
/*
*注意：當在運行以上代碼的時候編譯器會提示黃色波浪線的警告，這個是在提醒用戶沒有使用泛型
* 來限制集合的元素類型，這個問題需要用到後面的泛型集合來解決 這個章節裡無須理會
* 普通情況下集合都會把它的元素定義為對象類型，所以無論什麼類型都可以直接扔進集合中
*/
}
}
