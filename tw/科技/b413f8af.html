前言

你會因為寫文章而苦惱嗎？想不想利用Python的馬爾可夫模型幫助你自動生成文章，接下來看代碼吧

首先導入庫

import nltkimport random

主體代碼

file = open('Text/Walden.txt', 'r')walden = file.read()walden = walden.split()def makePairs(arr): pairs = [] for i in range(len(arr)): if i < len(arr) - 1: temp = (arr[i], arr[i + 1]) pairs.append(temp) return pairsdef generate(cfd, word='the', num=500): for i in range(num): # 用正確計數所示的單詞組成一個數組 arr = [] for j in cfd[word]: for k in range(cfd[word][j]): arr.append(j) print(word, end=' ') # 從條件分佈中隨機選擇單詞 word = arr[int((len(arr)) * random.random())]pairs = makePairs(walden)cfd = nltk.ConditionalFreqDist(pairs)generate(cfd)
