VBA｜使用窗體控件13：可摺疊對話框的RefEdit控件

2020-11-14
| 科技

使用RefEdit控件可摺疊當前操作的對話框，讓用戶在Excel工作表中拖動鼠標選擇一個單元格區域，並將該選擇區域填充到RefEdit控件框中。

下面以一個“用RefEdit實現的設置單元格格式窗體”的實例來說明選項按鈕的使用。

Private Sub UserForm_Initialize()

With cmbFont

.AddItem "宋體"

.AddItem "黑體"

.AddItem "楷體"

.AddItem "仿宋_GB2312"

.Value = .List(0)

End With

For i = 8 To 72

cmbSize.AddItem i

Next

cmbSize.Value = 8

End Sub

Private Sub chkBold_Click() '預覽字形(粗體)

lblPreview.Font.Bold = chkBold.Value

End Sub

Private Sub chkItalic_Click() '預覽字形(斜體)

lblPreview.Font.Italic = chkItalic.Value

End Sub

Private Sub cmbFont_Change() '預覽字體

lblPreview.Font.Name = cmbFont.Value

End Sub

Private Sub cmbSize_Change() '預覽字號

lblPreview.Font.Size = cmbSize.Value

End Sub

Private Sub cmdSet_Click()

Dim rng As Range, i As Integer, str1 As String

If RefEdit1.Value = "" Then '未選擇單元格區域

MsgBox "請先在工作表區域拖動鼠標，選擇一個單元格區域！", vbCritical + vbOKOnly

Exit Sub

End If

i = InStr(1, RefEdit1.Value, "!") '去掉工作表名

str1 = Mid(RefEdit1.Value, i + 1)

Set rng = Range(str1) '獲取對單元格區域的引用

With rng.Font '設置格式

.Name = cmbFont.Value '字體

.Size = cmbSize.Value '字號

.Bold = chkBold.Value '粗體

.Italic = chkItalic.Value '斜體

End With

End Sub

Private Sub cmdClose_Click()

Unload Me

End Sub
