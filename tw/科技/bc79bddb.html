(二)插入的模塊1 裡面的代碼：

Public mt As Table, c As Cell '定義mt為公有屬性表格對象變量,定義臨時c單元格為全局變量

Public i, j As Integer '定義公有屬性的計算結果單元格行i、列j號(格式如：1，3)

Sub Run_Calculate_Work() '運行計算工作

Dim rg As String, Error_Prompt As String 'rg為範圍字符串變量，Error_Prompt為輸入框內容是否為空或格式錯誤提示字符串變量

Retry:

Inputbox_title = "提示：“行號”介於2--" & mt.Rows.Count & "之間,“列號”介於1--" & mt.Columns.Count & "之間"

k = InputBox("請輸入結果單元格行i、列j號(格式如：1，3):", Inputbox_title)

'以下變量Error_Prompt是根據輸入對話框存於k中的內容判定獲取錯誤提示內容

Error_Prompt = IIf(Len(k) = 0, "為空！", "“" & k & "”格式錯誤！" & Chr(10) & "(參照輸入格式形如：1，3)")

If StrPtr(k) <> 0 Then

If InStr(k, "，") Or InStr(k, ",") Then

If InStr(k, "，") Then k = Replace(k, "，", ",") '將有可能輸入漢字標點的逗號"，"統統強制轉換為西文標點的逗號","

i = Val(Left(k, InStr(k, ",") - 1))

j = Val(Right(k, Len(k) - InStr(k, ",")))

If (i <= 1 Or i > mt.Rows.Count) Or (j < 1 Or j > mt.Columns.Count) Then

MsgBox "輸入內容" & Error_Prompt & Chr(10) & "請重新輸入！！！", vbInformation, "提示"

GoTo Retry

Else

If Not (c Is Nothing) Then Call Clear_Result_Before '根據對象是否為未指派具體實例的空對象，非空即刻清除原先的結果值

Set c = mt.Cell(i, j) '單元格對象c指向具體的mt表格對象的單元格Cell(i,j)

rg = "Cell(" & 1 & "," & 1 & ")" & ":" & "Cell(" & 1 & "," & 4 & ")"

MsgBox "計算區域是：" & rg, vbInformation, "提示"

c.Range.Text = ""

c.Shading.BackgroundPatternColor = wdColorAutomatic '將單元格c背景色填充顏色重置為無填充

Calc rg, c '調用自定義的求和過程

End If

Else

MsgBox "輸入內容" & Error_Prompt & Chr(10) & "請重新輸入！！！", vbInformation, "提示"

GoTo Retry

End If

Else

MsgBox "您點擊列<取消>按鈕或<X>關閉按鈕！" & Chr(10) & "因而也取消了計算區域數據的操作！", vbInformation, "提示"

End If

End Sub

Sub Clear_Result_Before()

If c Is Nothing Then

MsgBox "無結果數據，清除被禁止！", vbInformation, "提示"

Else

c.Range.Text = ""

c.Shading.BackgroundPatternColor = wdColorAutomatic

c.Shading.BackgroundPatternColor = wdColorBlue '重新將c單元格著色為藍色，以便下面消息框顯示後c單元格顏色暫留

MsgBox "清除計算結果單元格Cell(" & c.RowIndex & "," & c.ColumnIndex & ")的數據成功！", vbInformation, "提示"

c.Shading.BackgroundPatternColor = wdColorAutomatic '清除c單元格著色

Set c = Nothing '將定義的單元格變量對象重新恢復為空對象(即取消其指派的具體實例單元格)

End If

End Sub

Sub Calc(rg As String, c) '構造表格區域求和計算過程

Dim rg1 As String, rg2 As String, colstr As String, c1str As String, c2str As String

Dim r1 As Integer, c1 As Integer, r2 As Integer, c2 As Integer

colstr = "abcdefghijklmnopqrstuvwxyz" 'a--z單元格的列字符串，用於取出a--z中需要的列字符

If InStr(rg, "Cell") = 0 Or InStr(rg, ":") = 0 Or InStr(rg, ",") = 0 Then

MsgBox "error!"

Else

'rg計算數據區域範圍字符串分解為前後純的單元格的兩個子字符串rg1、rg2

rg1 = Left(rg, InStr(rg, ":") - 1)

rg2 = Right(rg, Len(rg) - InStr(rg, ":"))

r1 = Val(Mid(rg1, InStr(rg1, "(") + 1, Len(rg1) - InStr(rg1, ",") - 1)) '拆解出計算區域首單元格的行號

c1 = Val(Mid(rg1, InStr(rg1, ",") + 1, InStr(rg1, ")") - InStr(rg1, ",") - 1)) '拆解出計算區域首單元格的列號

r2 = Val(Mid(rg2, InStr(rg2, "(") + 1, Len(rg2) - InStr(rg2, ",") - 1)) '拆解出計算區域尾單元格的行號

c2 = Val(Mid(rg2, InStr(rg2, ",") + 1, InStr(rg2, ")") - InStr(rg2, ",") - 1)) '拆解出計算區域尾單元格的列號

'通過列查找字符串，獲取首尾單元格的“列”字符

c1str = Mid(colstr, c1, 1): c2str = Mid(colstr, c2, 1)

rg = c1str & r1 & ":" & c2str & r2 '重新生成的標準格式(形如A3:E8形式的範圍)字符串賦予變量rg以覆蓋原先的初始值

c.Formula "=sum(" & rg & ")" '單元格對象變量通過.Formula方法帶參數形式運用公式進行計算

c.Shading.BackgroundPatternColor = wdColorBlue '將單元格c背景色填充為藍色

End If

End Sub
