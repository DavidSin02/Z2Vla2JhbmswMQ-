C#算法系列（1）——二叉樹

2020-11-14
| 科技


本文主要講述二叉樹的兩種創建方式與四種遍歷方式，以及附上具體的實現代碼。

兩種創建方式分別為：順序存儲和鏈式存儲。遍歷方式有：先序遍歷、中序遍歷、後序遍歷以及層次遍歷。

一、 二叉樹的性質

介紹性質主要是為了後續進行編寫二叉樹遍歷算法的時會用到。

在二叉樹的第i層上至多有2^(i-1)個結點（i≥1）。

深度為k的二叉樹至多有2^k -1 個結點(k≥1)。

對任何一棵二叉樹T，如果其終端結點數為n0，度為2的結點數為n2，則n0 = n2 +1。

推導如下：

分支總數 = 總節點數 - 1（1）

分支總數 = 1*n1 + 2* n2 （n1表示度為1的結點數，n2表示度為2的結點數）（2）

總節點數 = n0 + n1 + n2 （3）

由（1），（2），（3）式即可得出n2 = n0 -1.

具有n個結點的完全二叉樹的深度為log2 (n) + 1。

如果對一棵有n個結點的完全二叉樹（其深度為log2 (n) + 1）的結點按層序編號，對任一結點i(1≤i≤n)有:

（1）如果i=1,則結點i是二叉樹的根，無雙親；如果i>1，則起雙親是結點i/2。

（2）如果2i>n，則結點i無左孩子。

（3）如果2i+1>n,則結點i無右孩子；否則其右孩子是結點2i+1。

二、二叉樹的構建

（1）順序存儲：用數組進行保存

（2）鏈式存儲：用兩塊指針域進行保存，主要思想是採用雙向鏈表的思想

三、二叉樹的具體實現

(1)順序存儲的具體實現：

using System;

using System.Collections.Generic;

using System.Linq;

using System.Text;

using System.Threading.Tasks;

namespace 二叉樹_順序存儲

{

class BinaryTree<T>

{

private T[] data;

private int count = 0;

public BinaryTree(int capacity)

{

data = new T[capacity];

}

public bool Add(T item)

{

if (count >= data.Length)

return false;

data[count] = item;

count++;

return true;

}

public void FirstTraversal()

{

FirstTraversal(0);

}

private void FirstTraversal(int index)

{

if (index >= count || data[index].Equals(-1))

return;

int number = index + 1;

Console.Write(data[index] + " ");

int leftNumber = number * 2;

int rightNumber = number * 2 + 1;

FirstTraversal(leftNumber-1);

FirstTraversal(rightNumber-1);

}

public void MiddleTraversal()

{

MiddleTraversal(0);

}

private void MiddleTraversal(int index)

{

if (index >= count || data[index].Equals(-1))

return;

int number = index + 1;

int leftNumber = number * 2;

int rightNumber = number * 2 + 1;

MiddleTraversal(leftNumber - 1);

Console.Write(data[index] + " ");

MiddleTraversal(rightNumber - 1);

}

public void LastTraversal()

{

LastTraversal(0);

}

private void LastTraversal(int index)

{

if (index >= count|| data[index].Equals(-1))

return;

int number = index + 1;

int leftNumber = number * 2;

int rightNumber = number * 2 + 1;

LastTraversal(leftNumber - 1);

LastTraversal(rightNumber - 1);

Console.Write(data[index] + " ");

}

public void LayerTraversal()

{

for (int i = 0; i < count; i++)

{

if (data[i].Equals(-1))

continue;

Console.Write(data[i] + " ");

}

}

}

}

（2）鏈式存儲具體實現方式：

首先需要定義一個二叉樹的結點類TreeNode.cs，具體代碼如下：

using System;

using System.Collections.Generic;

using System.Linq;

using System.Text;

using System.Threading.Tasks;

namespace 二叉樹_鏈式存儲

{

class TreeNode<T>

{

private T data;

private TreeNode<T> lChild;

private TreeNode<T> rChild;

public T Data

{

get { return data; }

set { data = value; }

}

public TreeNode<T> LChild

{

get { return lChild; }

set { lChild = value; }

}

public TreeNode<T> RChild

{

get { return rChild; }

set { rChild = value;}

}

public TreeNode()

{

data = default(T);

lChild = null;

rChild = null;

}

public TreeNode(T val)

{

data = val;

lChild = null;

rChild = null;

}

}

}

再定義一個二叉樹類BinaryTree.cs

using System;

using System.Collections.Generic;

using System.Linq;

using System.Text;

using System.Threading.Tasks;

namespace 二叉樹_鏈式存儲

{

class BinaryTree<T>

{

private TreeNode<T> head;

private T[] datas;

public TreeNode<T> Head

{

get { return head; }

}

public BinaryTree(T[] vals)

{

datas = vals;

Add(head,0);

}

private void Add(TreeNode<T> parent, int index)

{

if (parent == null)

{

parent = new TreeNode<T>(datas[index]);

head = parent;

}

int leftIndex = 2 * index + 1;

int rightIndex = 2 * index + 2;

if (leftIndex < datas.Length)

{

if (!datas[leftIndex].Equals("#"))

{

parent.LChild = new TreeNode<T>(datas[leftIndex]);

Add(parent.LChild, leftIndex);

}

else

{

parent.LChild = null;

}

}

if (rightIndex < datas.Length)

{

if (!datas[rightIndex].Equals("#"))

{

parent.RChild = new TreeNode<T>(datas[rightIndex]);

Add(parent.RChild, rightIndex);

}

else

{

parent.RChild = null;

}

}

}

public void PreTraversal(TreeNode<T> node)

{

if (node != null)

{

Console.Write(node.Data + " ");

PreTraversal(node.LChild);

PreTraversal(node.RChild);

}

}

public void InTraversal(TreeNode<T> node)

{

if (node != null)

{

InTraversal(node.LChild);

Console.Write(node.Data + " ");

InTraversal(node.RChild);

}

}

public void LastTraversal(TreeNode<T> node)

{

if (node != null)

{

LastTraversal(node.LChild);

LastTraversal(node.RChild);

Console.Write(node.Data + " ");

}

}

public void LevelTranversal(TreeNode<T> node)

{

if (node == null)

{

return;

}

Queue<TreeNode<T>> queue = new Queue<TreeNode<T>>();

queue.Enqueue(node);

while (queue.Count > 0)

{

TreeNode<T> temp = queue.Dequeue();

Console.Write(temp.Data+" ");

if (temp.LChild != null)

{

queue.Enqueue(temp.LChild);

}

if (temp.RChild != null)

{

queue.Enqueue(temp.RChild);

}

}

}

}

}

using System

using System.Collections.Generic

using System.Linq

using System.Text

using System.Threading.Tasks

namespace 二叉樹_鏈式存儲

{

class Program

{

{

Console.Write("先序遍歷：")

tree.InTraversal(tree.Head)

Console.WriteLine()

tree.LevelTranversal(tree.Head)

Console.ReadKey()

}

}

}


以上就是二叉樹的順序存儲和鏈式存儲方式，順序存儲按照數組來保存，鏈式按照雙向鏈表的思想進行存儲。在鏈式存儲過程中，葉子結點會存在大量的空指針域的情況，因此引出了後續的線索二叉樹。
