多進程併發服務器

使用多進程併發服務器時要考慮以下幾點：

1. 父進程最大文件描述個數(父進程中需要close關閉accept返回的新文件描述符)

2. 系統內創建進程個數(與內存大小相關)

3. 進程創建過多是否降低整體服務性能(進程調度)

客戶端 client

#include<unistd.h>

#include<stdio.h>

#include<arpa/inet.h>

#include<sys/types.h>

#include<sys/socket.h>

#include<netinet/in.h>

#include<string.h>

#include<strings.h>

#include"wrap.h"

#define SERV_PORT 8000

#define MAXLINE 8192

int main(void)

{

int sfd;

struct sockaddr_in serv_addr;

char buf[MAXLINE];

int n;

// 創建和服務端通信的套接字

sfd = Socket(AF_INET,SOCK_STREAM,0);

// 給 socket 結構體清零

bzero(&serv_addr,sizeof(serv_addr));

// 給 serv_addr 結構體初始化

serv_addr.sin_family = AF_INET;

serv_addr.sin_port = htons(SERV_PORT);

inet_pton(AF_INET,"127.0.0.1",&serv_addr.sin_addr.s_addr);

// 連接服務器端

Connect(sfd,(struct sockaddr*)&serv_addr,sizeof(serv_addr));

while(fgets(buf,MAXLINE,stdin) != NULL)

{

// 把從鍵盤讀到的數據發送給服務器

Write(sfd,buf,strlen(buf));

// 從服務器端接收數據

n = Read(sfd,buf,MAXLINE);

if( n == 0 )

{

printf("the other side has been closed\n");

break;

}

else

Write(STDOUT_FILENO,buf,strlen(buf));

}

return 0;

}

多線程併發服務器

在使用線程模型開發服務器時需考慮以下問題：

1. 調整進程內最大文件描述符上限

2. 線程如有共享數據，考慮線程同步

3. 服務於客戶端線程退出時，退出處理。（退出值，分離態）

4. 系統負載，隨著鏈接客戶端增加，導致其它線程不能及時得到CPU

服務端 server

/*

多線程 實現 高併發服務器

*/

#include<unistd.h>

#include<pthread.h>

#include<arpa/inet.h>

#include<sys/types.h>

#include<stdio.h>

#include<string.h>

#include<strings.h>

#include<stdlib.h>

#include<ctype.h>

#include"wrap.h"

#define MAXLINE 8192

#define SERV_PORT 8000

// 定義一個結構體，將地址結構和 cfd 綁定在一起

struct s_info

{

struct sockaddr_in cliaddr;

int cfd;

};

void *do_work(void *arg)

{

int n,i;

struct s_info *ts = (struct s_info*)arg;

char buf[MAXLINE];

char str[INET_ADDRSTRLEN]; // #define INET_ADDRSTRLEN 16 可用 [+d 來查看

while(1)

{

// 讀客戶端

n = Read(ts->cfd,buf,MAXLINE);

if( n == 0 )

{

printf("the client %d closed ...\n",ts->cfd);

break; // 退出循環，關閉 cfd

}

// 打印客戶端信息(IP/PORT)

printf("recieve from %s at prot %d\n",inet_ntop(AF_INET,&(*ts).cliaddr.sin_addr,str,sizeof(str)),

ntohs((*ts).cliaddr.sin_port));

for(i=0;i<n;i++)

buf[i] = toupper(buf[i]);


// 把數據寫到屏幕上

Write(STDOUT_FILENO,buf,n);

// 把轉換後的數據發送給客戶端

Write(ts->cfd,buf,n);

}

// 關閉和客戶端連接的套接字

Close(ts->cfd);

return (void*)0;

}

int main(void)

{

struct sockaddr_in serveraddr,cliaddr;

socklen_t cliaddr_len;

int listenfd,connfd;

pthread_t tid;

struct s_info ts[256]; // 根據最大線程數，創建結構體數組

int i = 0;

// 創建一個 socket 得到 lfd;

listenfd = Socket(AF_INET,SOCK_STREAM,0);

// socket 通信的數據結構體 serveraddr 清空

bzero(&serveraddr,sizeof(serveraddr));

// 初始化地址結構體

serveraddr.sin_family = AF_INET; // 規定通信協議為 IPV4

serveraddr.sin_port = htons(SERV_PORT); // 指定端口號為 8000

serveraddr.sin_addr.s_addr = htonl(INADDR_ANY); // 指定本地任意 IP 地址

Bind(listenfd,(struct sockaddr*)&serveraddr,sizeof(serveraddr)); // 綁定

Listen(listenfd,128); // 設置同一時刻連接服務器的上限數

printf("Accept client connect ...\n");

while(1)

{

cliaddr_len = sizeof(cliaddr);

// 阻塞監聽客戶端請求

connfd = Accept(listenfd,(struct sockaddr*)&cliaddr,&cliaddr_len);

ts[i].cliaddr = cliaddr;

ts[i].cfd = connfd;

// 達到線程最大數時，pthread_create 出錯處理，增加服務器的穩定性

pthread_create(&tid,NULL,do_work,(void*)&ts[i]);

pthread_detach(tid); // 子線程分離，防止殭屍線程產生

i++;

}

return 0;

}

注：

為了減少進程服務器和多線服務器端代碼的冗餘性。我把出錯處理全部封裝成了獨立的函數。把 socket、bind、listen、accept、connect、close、read、write 等函數進行了封裝，把所有出錯，封裝在了大寫的 Socket、Bind、Listen、Accept、Connect、Close、Read、Write。因此，在服務端和客戶端，可以不用進行錯誤處理。而且，還有另外一個好處，把以上的函數封裝成大寫的函數，在程序中，當光標在 大寫函數上時，按下 shift + k ，即跳轉到小寫的 man page 中對應的函數詳解。

int n;

again:

// 參數三：應該讀取的字節數

ssize_t Readn(int fd,void *vptr,size_t n)

{

size_t nleft; // unsigned int 剩餘未讀取的字節數

ssize_t nread; // int 實際讀到的字節數

char *ptr;

ptr = vptr;

nleft = n;

while(nleft>0)

{

if(( nread = read(fd,ptr,nleft)) <0 ) // 如果讀數據時出錯

{

if( errno == EINTR ) //被信號打斷

nread = 0;

else // 其他出錯

return -1;

}

else if( nread == 0 ) // 如果數據讀完的話，跳出循環

break;

else

{

nleft -= nread;

ptr += nread;

}

}

return n - nleft;

}

ssize_t Writen(int fd,const void*vptr,size_t n)

{

size_t nleft;

ssize_t nwritten;

const char* ptr;

nleft = n;

ptr = vptr;

while(nleft>0)

{

if( (nwritten = write(fd,ptr,nleft))<0 )

{

if( nwritten <0 && errno == EINTR )

nwritten = 0;

else

return -1;

}

nleft -= nwritten;

ptr +=nwritten;

}

return n;

}

static ssize_t my_read(int fd,char *ptr)

{

static int read_cnt;

static char* read_ptr;

static char read_buf[100];

if( read_cnt <= 0 )

{

again:

if( (read_cnt = read(fd,read_buf,sizeof(read_buf))) < 0 )

{

if( errno == EINTR )

goto again;

return -1;

}

else if( read_cnt == 0 )

return 0;

read_ptr = read_buf;

}

read_cnt--;

*ptr = *read_ptr++;

return 1;

}

ssize_t Readline(int fd,void *vptr,ssize_t maxlen)

{

ptr = vptr;

for( n=1;n<maxlen;n++)

{

if( (rc = my_read(fd,&c)) == 1 )

{

*ptr++ = c;

break;

}

else if( rc == 0 )

{

*ptr = 0;

return n-1;

}

else

return -1;

}

*ptr = 0;

return n;

}
