網易音樂統計歌單熱度+歌單名詞頻，並可視化

2020-10-29
| 科技

  • from selenium import webdriver
  • import time
  • import MySQLdb
  • from bs4 import BeautifulSoup
  • import re
  • from pyecharts import Bar
  • from pyecharts import WordCloud
  • from pyecharts import Page
  • from wordcloud import WordCloud,ImageColorGenerator
  • import matplotlib.pyplot as plt
  • import jieba
  • import sys
  • reload(sys)
  • sys.setdefaultencoding('utf8')
  • '''
  • #連接數據庫
  • conn=MySQLdb.connect("localhost","root","1204","wymusicdata" ,charset="utf8")
  • cur=conn.cursor()
  • print "連接數據庫成功"
  • '''
  • x=webdriver.Chrome() #不能少了括號
  • page=0
  • listenNums=[]
  • songLists=[]
  • authors=[]
  • #網易雲音樂網頁採用了框架，需要爬取的歌單在iframe裡面，
  • # 這種情況我們直接定位是不能定位到元素，
  • # 採用x.switch_to_frame（“”）進入框架進行數據爬取
  • while True:
  • x.get('http://music.163.com/discover/playlist/?offset=%s' % page)
  • #x.get('http://music.163.com/discover/playlist/?offset=210')
  • x.switch_to_frame(x.find_element_by_name("contentFrame"))
  • html=x.page_source
  • soup=BeautifulSoup(html,"html.parser")
  • #精確定位找到我們需要的<li>標籤內容
  • cm=soup.find(class_="m-cvrlst f-cb")
  • contents=cm.find_all('li')
  • for content in contents:
  • wan=str(content)
  • if "萬" in wan:
  • wan=wan.replace('萬',"0000")
  • else:
  • wan=wan
  • listenNum=re.findall(r'<span class="nb">(.*?)</span>',str(wan))
  • #songList=re.findall(r'<a class="tit f-thide s-fc0" href=".*?" title="(".*")">',str(content))
  • songList=content.find(class_="tit f-thide s-fc0").text
  • #author=re.findall(r'<a class="nm nm-icn f-thide s-fc3" href=".*?" title=".*?">(.*?)</a>',str(content))
  • author=content.find(class_="nm nm-icn f-thide s-fc3").text
  • #取出來的元素放進列表
  • listenNums.append(listenNum[0])
  • songLists.append(songList)
  • authors.append(author)

  • #到瀏覽到最後一頁時("js-disabled")=下一頁無法點擊 即認為 爬取完成
  • if page!=0:
  • try:
  • x.find_element_by_class_name("js-disabled")
  • print "全部歌單爬取成功"
  • break
  • except:
  • print "已成功爬取第%s頁"% (page/35+1)
  • pass

  • page+=35
  • # if page==70:
  • # break
  • time.sleep(2)
  • #由於網易雲音樂網站在底部有個播放器的小元素導致無法點擊下一頁
  • # try:
  • # x.find_element_by_class_name("znxt").click()
  • # time.sleep(2)
  • # except:
  • # print "全部歌單爬取成功"
  • # break
  • #在這裡只能採用分析歌單的url，遍歷所有的連接來代替點擊下一頁
  • #--------------------------------------------

  • #歌單名字做詞頻統計繪製詞雲圖
  • ciyun=''
  • for words in songLists:
  • #rfw=re.match("([\u4e00-\u9fa5])",words)
  • pattern =re.compile(u"[\u4e00-\u9fa5]+")
  • result=re.findall(pattern,words)
  • for r in result:
  • word=jieba.cut(r)
  • if ciyun!='':
  • ciyun=ciyun+','+",".join(word)
  • else:
  • ciyun+=",".join(word)
  • ciyunList=ciyun.split(",")
  • dict={}
  • for i in ciyunList:
  • if i not in dict:
  • dict[i]=1
  • else:
  • dict[i]+=1
  • key=[]
  • value=[]
  • for k,v in dict.items():
  • key.append(k)
  • value.append(v)

  • #page=Page()
  • bar=Bar("網易Music","網易雲音樂歌單熱度")
  • bar.add("歌單熱度",songLists,listenNums,mark_point=["max", "min"])
  • #page.add(bar)
  • bar.render()
  • '''
  • wordcloud=WordCloud("歌單名詞雲",width=600,height=800)
  • wordcloud.add("",key,value,word_size_range=[20,100],shape="cardioid")
  • page.add(wordcloud)
  • #page.render()
  • '''
  • #詞雲
  • bgImage=plt.imread('xin.jpg')
  • wc=WordCloud(background_color='black',
  • mask=bgImage,
  • font_path='simkai.ttf',
  • max_font_size=50,
  • random_state=30,
  • )
  • wc.generate(ciyun)
  • plt.imshow(wc)
  • wc.to_file("WYsongLciyun.png")
